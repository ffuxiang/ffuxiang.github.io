    class VOCDataSet(Dataset):
    """读取解析PASCAL VOC2007/2012数据集"""

    def __init__(self, voc_root, year="2012", transforms=None, txt_name: str = "train.txt"):
        assert year in ["2007", "2012"], "year must be in ['2007', '2012']"
        # 得到数据集根路径
        self.root = os.path.join(voc_root, "VOCdevkit", f"VOC{year}")
        # 图像根目录，存有所有图像的目录
        self.img_root = os.path.join(self.root, "JPEGImages")
        # 注释根目录，存有所有xml文件的目录
        self.annotations_root = os.path.join(self.root, "Annotations")

        # read train.txt or val.txt file，读取存有所有图片名称的txt文件
        txt_path = os.path.join(self.root, "ImageSets", "Main", txt_name)
        assert os.path.exists(txt_path), "not found {} file.".format(txt_name)
        # 遍历读取到的txt文件
        with open(txt_path) as read:
            # .strip()去掉换行符，这里读取所有xml的地址，储存成一个列表
            self.xml_list = [os.path.join(self.annotations_root, line.strip() + ".xml")
                             for line in read.readlines() if len(line.strip()) > 0]

        # check file，xml个数必须大于0
        assert len(self.xml_list) > 0, "in '{}' file does not find any information.".format(txt_path)
        for xml_path in self.xml_list:
            # 检查每个xml地址是否存在
            assert os.path.exists(xml_path), "not found '{}' file.".format(xml_path)
        # read class_indict
        # 读取类别信息
        json_file = './pascal_voc_classes.json'
        # 检查储存类别的json文件是否存在
        assert os.path.exists(json_file), "{} file not exist.".format(json_file)
        json_file = open(json_file, 'r')
        # 得到类别列表
        self.class_dict = json.load(json_file)
        json_file.close()
        # 数据预处理操作
        self.transforms = transforms

    def __len__(self):
        return len(self.xml_list)

    def __getitem__(self, idx):
        # read xml，根据序号idx，得到xml路径
        xml_path = self.xml_list[idx]
        # 打开文件
        with open(xml_path) as fid:
            xml_str = fid.read()
        xml = etree.fromstring(xml_str)
        # 得到字典格式的标签数据
        data = self.parse_xml_to_dict(xml)["annotation"]
        # 得到图片路径
        img_path = os.path.join(self.img_root, data["filename"])
        # 读取图片
        image = Image.open(img_path)
        # 规定图片格式
        if image.format != "JPEG":
            raise ValueError("Image '{}' format not JPEG".format(img_path))
        # 初始化存储标签信息的变量
        boxes = []
        labels = []
        iscrowd = []
        assert "object" in data, "{} lack of object information.".format(xml_path)
        for obj in data["object"]:
            # 得到边界框坐标
            xmin = float(obj["bndbox"]["xmin"])
            xmax = float(obj["bndbox"]["xmax"])
            ymin = float(obj["bndbox"]["ymin"])
            ymax = float(obj["bndbox"]["ymax"])

            # 进一步检查数据，有的标注信息中可能有w或h为0的情况，这样的数据会导致计算回归loss为nan
            if xmax <= xmin or ymax <= ymin:
                print("Warning: in '{}' xml, there are some bbox w/h <=0".format(xml_path))
                continue
            # 存储边界框坐标信息
            boxes.append([xmin, ymin, xmax, ymax])
            # 得到类别
            labels.append(self.class_dict[obj["name"]])
            # difficult为1表示检测比较难，为0表示检测比较容易
            if "difficult" in obj:
                iscrowd.append(int(obj["difficult"]))
            else:
                iscrowd.append(0)

        # 将所有数据转化为tensor格式
        boxes = torch.as_tensor(boxes, dtype=torch.float32)
        labels = torch.as_tensor(labels, dtype=torch.int64)
        iscrowd = torch.as_tensor(iscrowd, dtype=torch.int64)
        image_id = torch.tensor([idx])
        # 计算边界框区域面积
        area = (boxes[:, 3] - boxes[:, 1]) * (boxes[:, 2] - boxes[:, 0])
        # 将标签数据转化为字典格式，便于后续调用
        target = {}
        target["boxes"] = boxes
        target["labels"] = labels
        target["image_id"] = image_id
        target["area"] = area
        target["iscrowd"] = iscrowd
        # 预处理操作
        if self.transforms is not None:
            # target也需要做transforms
            image, target = self.transforms(image, target)
        # 返回读到的图片和相应的标签
        return image, target

    将xml文件解析成字典形式的数据

    def parse_xml_to_dict(self, xml):
    """
    将xml文件解析成字典形式，参考tensorflow的recursive_parse_xml_to_dict
    Args:
        xml: xml tree obtained by parsing XML file contents using lxml.etree

    Returns:
        Python dictionary holding XML contents.
    """
    # 如果遍历到底层，直接返回tag对应的信息
    if len(xml) == 0:
        return {xml.tag: xml.text}

    result = {}
    for child in xml:
        # 递归遍历标签信息
        child_result = self.parse_xml_to_dict(child)
        if child.tag != 'object':
            # 如果不是object信息，则直接赋值即可，默认自动创建字典
            result[child.tag] = child_result[child.tag]
        else:
            # 因为object可能有多个，所以需要放入列表里
            # 如果字典不存在object键，则新建一个
            if child.tag not in result:
                result[child.tag] = []
            # 传入object列表中
            result[child.tag].append(child_result[child.tag])
    # 返回字典格式的标签
    return {xml.tag: result}

    将训练集图片按高宽比例划分，在同一批次中只采相同区间内的图片

    class GroupedBatchSampler(BatchSampler):
    """
    Wraps another sampler to yield a mini-batch of indices.
    It enforces that the batch only contain elements from the same group.
    It also tries to provide mini-batches which follows an ordering which is
    as close as possible to the ordering from the original sampler.
    Arguments:
        sampler (Sampler): Base sampler.
        group_ids (list[int]): If the sampler produces indices in range [0, N),
            `group_ids` must be a list of `N` ints which contains the group id of each sample.
            The group ids must be a continuous set of integers starting from
            0, i.e. they must be in the range [0, num_groups).
        batch_size (int): Size of mini-batch.
    """
    def __init__(self, sampler, group_ids, batch_size):
        if not isinstance(sampler, Sampler):
            raise ValueError(
                "sampler should be an instance of "
                "torch.utils.data.Sampler, but got sampler={}".format(sampler)
            )
        # 随机采样器
        self.sampler = sampler
        # 图像分组索引
        self.group_ids = group_ids
        # batch size
        self.batch_size = batch_size

    def __iter__(self):
        # 初始化字典，值的数据类型为列表格式
        buffer_per_group = defaultdict(list)
        samples_per_group = defaultdict(list)

        num_batches = 0
        # 随机得到一个序号
        for idx in self.sampler:
            # 得到被采样图像的区间分组情况
            group_id = self.group_ids[idx]
            # 键为分组标签，值为图片序号组成的列表
            buffer_per_group[group_id].append(idx)
            samples_per_group[group_id].append(idx)
            # 如果当前采样图片所在的区间中，图片总数已经达到batch size大小，即满足一个批次
            if len(buffer_per_group[group_id]) == self.batch_size:
                # 则将该区间目前的图片打包成一个可迭代的对象，后面会进入训练过程中的for循环
                yield buffer_per_group[group_id]
                # batch数量加一
                num_batches += 1
                # 之后删去该区间中已经训练过的图片
                del buffer_per_group[group_id]
            assert len(buffer_per_group[group_id]) < self.batch_size

        """
        目前已经将所有图片遍历完了，但是每个区间中可能还留有部分图片未参加训练(数目不够构成一个batch)
        因此batch的总数达不到期望的要求，这里需要对部分组中的图片做扩充，利用之前训练过的图片来扩充当前组内的图片
        从而构成一个batch来训练网络。
        这里扩充组内剩余图片数量多的区间分组，并不是对所有分组都扩充，只要batch数量满足期望的数量就可以
        """

        # 期望的batch数量
        expected_num_batches = len(self)
        # 期望的减去已经得到的，这里得到还差多少batch
        num_remaining = expected_num_batches - num_batches
        if num_remaining > 0:
            # 按目前每个区间内剩余图片的数量来遍历，先扩充组内图片数量剩余多的分组
            for group_id, _ in sorted(buffer_per_group.items(),
                                      key=lambda x: len(x[1]), reverse=True):
                # remaining表示当前分组还差多少张图片可以组成一个batch
                remaining = self.batch_size - len(buffer_per_group[group_id])
                # 这里表示将之前训练过的数据复制几遍，从而保证samples_from_group_id中至少有remaining个数据
                samples_from_group_id = _repeat_to_at_least(samples_per_group[group_id], remaining)
                # .extend表示扩展原列表，直接选取前remaining个id即可
                buffer_per_group[group_id].extend(samples_from_group_id[:remaining])
                # 再次检查是否可以构成一个batch size
                assert len(buffer_per_group[group_id]) == self.batch_size
                # 打包成一个batch，参与训练
                yield buffer_per_group[group_id]
                num_remaining -= 1
                # 如果batch数量达到期望的数量了，则跳出循环
                if num_remaining == 0:
                    break
        assert num_remaining == 0

    def __len__(self):
        # batch总数，样本数除以batch size
        return len(self.sampler) // self.batch_size

    重复足够多的次数

    def _repeat_to_at_least(iterable, n):
    # math.ceil表示上取整操作
    repeat_times = math.ceil(n / len(iterable))
    # repeat表示将iterable重复repeat_times遍，之后利用chain.from_iterable转化为可迭代的对象
    repeated = chain.from_iterable(repeat(iterable, repeat_times))
    # 利用list方法转化为list格式的数据
    # 相当于复制repeat_times遍iterable，之后再将复制的列表拼接起来
    return list(repeated)

    按宽高比例对图片分组¶

    def create_aspect_ratio_groups(dataset, k=0):
    # 计算所有数据集中的图片width/height比例
    aspect_ratios = compute_aspect_ratios(dataset)
    # 将[0.5, 2]区间划分成2*k等份，2k+1个点，这里划成2k个区间，后续再加上0到0.5，与2到inf(正无穷)，一共得到2(k+1)个区间
    bins = (2 ** np.linspace(-1, 1, 2 * k + 1)).tolist() if k > 0 else [1.0]

    # 统计所有图像比例在bins区间中的位置索引，返回的是区间索引
    groups = _quantize(aspect_ratios, bins)
    # count number of elements per group
    # 统计每个区间的频次，即得到每个区间中的图像数量
    counts = np.unique(groups, return_counts=True)[1]
    # 区间划分情况，从0到正无穷
    fbins = [0] + bins + [np.inf]
    print("Using {} as bins for aspect ratio quantization".format(fbins))
    print("Count of instances per bin: {}".format(counts))
    return groups

    计算宽高比例¶

    def compute_aspect_ratios(dataset, indices=None):
    # hasattr函数用于判断对象是否包含对应的属性，主要用到这个分支
    if hasattr(dataset, "get_height_and_width"):
        return _compute_aspect_ratios_custom_dataset(dataset, indices)

def _compute_aspect_ratios_custom_dataset(dataset, indices=None):
    if indices is None:
        indices = range(len(dataset))
    # 初始化储存宽高比例的变量
    aspect_ratios = []
    # 遍历所有图片
    for i in indices:
        # 利用定义好的get_height_and_width来获取图片高、宽
        height, width = dataset.get_height_and_width(i)
        # 计算比例
        aspect_ratio = float(width) / float(height)
        aspect_ratios.append(aspect_ratio)
    # 返回得到的比例
    return aspect_ratios

    # 计算图片高宽，用于后续计算图片高宽比例，针对比例来采样进行训练
def get_height_and_width(self, idx):
    # 读取xml文件
    xml_path = self.xml_list[idx]
    with open(xml_path) as fid:
        xml_str = fid.read()
    xml = etree.fromstring(xml_str)
    # 转化为字典格式
    data = self.parse_xml_to_dict(xml)["annotation"]
    # 直接读取size中的高、宽数值
    data_height = int(data["size"]["height"])
    data_width = int(data["size"]["width"])
    # 依次返回高、宽
    return data_height, data_width

    按图片宽高比例与区间节点分组

    def _quantize(x, bins):
    bins = copy.deepcopy(bins)
    # 将划分区间的点进行排序
    bins = sorted(bins)
    # bisect_right：寻找y元素按顺序应该排在bins中哪个元素的右边，返回的是区间索引
    # 这里相当于做了一个遍历，遍历x中的所有元素，得到每个x应该位于哪个区间
    quantized = list(map(lambda y: bisect.bisect_right(bins, y), x))
    return quantized

    训练过程¶

    def main(parser_data):
    # 如果显卡可用，则使用GPU训练，否则使用CPU训练
    device = torch.device(parser_data.device if torch.cuda.is_available() else "cpu")
    print("Using {} device training.".format(device.type))

    # 用来保存训练日志
    results_file = "results{}.txt".format(datetime.datetime.now().strftime("%Y%m%d-%H%M%S"))
    # 数据预处理操作，训练阶段多一个随机水平翻转操作
    data_transform = {
        "train": transforms.Compose([transforms.ToTensor(),
                                     transforms.RandomHorizontalFlip(0.5)]),
        "val": transforms.Compose([transforms.ToTensor()])
    }
    # 数据根目录
    VOC_root = parser_data.data_path
    # 检查该目录是否存在
    if os.path.exists(os.path.join(VOC_root, "VOCdevkit")) is False:
        raise FileNotFoundError("VOCdevkit dose not in path:'{}'.".format(VOC_root))

    # 加载数据集
    # VOCdevkit -> VOC2012 -> ImageSets -> Main -> train.txt
    train_dataset = VOCDataSet(VOC_root, "2012", data_transform["train"], "train.txt")
    train_sampler = None

    # 是否按图片相似高宽比采样图片组成batch
    # 使用的话能够减小训练时所需GPU显存，默认使用
    if args.aspect_ratio_group_factor >= 0:
        # 先初始化一个随机的采样器
        train_sampler = torch.utils.data.RandomSampler(train_dataset)
        # 统计所有图像高宽比例在bins区间中的位置索引
        # args.aspect_ratio_group_factor默认为3，表示将图像数据的高宽比例划分成2(k+1)个区间
        group_ids = create_aspect_ratio_groups(train_dataset, k=args.aspect_ratio_group_factor)
        # 每批batch在采样图片时，均从同一高宽比例的区间中采样
        train_batch_sampler = GroupedBatchSampler(train_sampler, group_ids, args.batch_size)

    # batch size
    batch_size = parser_data.batch_size
    # 得到cpu线程，注意这里的线程数量必须小于等于batch size的大小
    nw = min([os.cpu_count(), batch_size if batch_size > 1 else 0, 8])  # number of workers
    print('Using %g dataloader workers' % nw)
    if train_sampler:
        # 如果按照图片高宽比采样图片，dataloader中需要使用batch_sampler，即之前定义好的采样策略
        # 注意这里的collate_fn是自定义的，因为读取的数据包括image和targets，不能直接使用默认的方法合成batch
        train_data_loader = torch.utils.data.DataLoader(train_dataset,
                                                        batch_sampler=train_batch_sampler,
                                                        pin_memory=True,
                                                        num_workers=nw,
                                                        collate_fn=train_dataset.collate_fn)
    else:
        # 如果不按长宽比例相似度采样的话，则默认使用随机采样策略
        train_data_loader = torch.utils.data.DataLoader(train_dataset,
                                                        batch_size=batch_size,
                                                        shuffle=True,
                                                        pin_memory=True,
                                                        num_workers=nw,
                                                        collate_fn=train_dataset.collate_fn)

    # 加载测试数据
    # VOCdevkit -> VOC2012 -> ImageSets -> Main -> val.txt
    val_dataset = VOCDataSet(VOC_root, "2012", data_transform["val"], "val.txt")
    # 加载测试数据由于是一张一张进行测试，因此不需要修改采样策略
    val_data_set_loader = torch.utils.data.DataLoader(val_dataset,
                                                      batch_size=1,
                                                      shuffle=False,
                                                      pin_memory=True,
                                                      num_workers=nw,
                                                      collate_fn=val_dataset.collate_fn)

    # 创建模型，注意这里分类数要+1，序号为0时表示背景
    model = create_model(num_classes=parser_data.num_classes + 1)
    # 将模型放入特定的储存设备
    model.to(device)

    # 定义优化器
    params = [p for p in model.parameters() if p.requires_grad]
    optimizer = torch.optim.SGD(params, lr=0.005,
                                momentum=0.9, weight_decay=0.0005)
    # 梯度缩放，防止出现梯度下溢的现象，由于存储小数点的位数有限，如果梯度过小则容易四舍五入得到0
    # 这里就对每个梯度乘上一个缩放因子，放大梯度从而防止下溢，默认不用梯度缩放
    scaler = torch.cuda.amp.GradScaler() if args.amp else None

    # 学习率衰减策略
    lr_scheduler = torch.optim.lr_scheduler.StepLR(optimizer,
                                                   step_size=3,
                                                   gamma=0.33)

    # 如果指定了上次训练保存的权重文件地址，则接着上次结果接着训练
    if parser_data.resume != "":
        # 加载模型参数、优化器等参数
        checkpoint = torch.load(parser_data.resume, map_location='cpu')
        model.load_state_dict(checkpoint['model'])
        optimizer.load_state_dict(checkpoint['optimizer'])
        lr_scheduler.load_state_dict(checkpoint['lr_scheduler'])
        parser_data.start_epoch = checkpoint['epoch'] + 1
        if args.amp and "scaler" in checkpoint:
            scaler.load_state_dict(checkpoint["scaler"])
        print("the training process from epoch{}...".format(parser_data.start_epoch))

    train_loss = []
    learning_rate = []
    val_map = []
    # 开始遍历数据，进行训练
    for epoch in range(parser_data.start_epoch, parser_data.epochs):
        # 训练一个epoch
        # 返回平均损失和当前的学习率
        mean_loss, lr = utils.train_one_epoch(model, optimizer, train_data_loader,
                                              device=device, epoch=epoch,
                                              print_freq=50, warmup=True,
                                              scaler=scaler)
        # 记录训练损失
        train_loss.append(mean_loss.item())
        # 记录学习率
        learning_rate.append(lr)
        # 更新学习率(这里与学习率下降策略有关)
        lr_scheduler.step()
        # 测试模型
        coco_info = utils.evaluate(model, val_data_set_loader, device=device)

        # 将训练日志写入txt文件
        with open(results_file, "a") as f:
            # 写入的数据包括coco指标还有loss和learning rate
            result_info = [str(round(i, 4)) for i in coco_info + [mean_loss.item()]] + [str(round(lr, 6))]
            txt = "epoch:{} {}".format(epoch, '  '.join(result_info))
            f.write(txt + "\n")
        # 更新模型mAP数据
        val_map.append(coco_info[1])  # pascal mAP

        # save weights，保存各种权重参数，模型、优化器、学习率
        save_files = {
            'model': model.state_dict(),
            'optimizer': optimizer.state_dict(),
            'lr_scheduler': lr_scheduler.state_dict(),
            'epoch': epoch}
        if args.amp:
            save_files["scaler"] = scaler.state_dict()
        torch.save(save_files, "./save_weights/resNetFpn-model-{}.pth".format(epoch))

    # 损失、学习率折线图
    if len(train_loss) != 0 and len(learning_rate) != 0:
        from plot_curve import plot_loss_and_lr
        plot_loss_and_lr(train_loss, learning_rate)

    # mAP折线图
    if len(val_map) != 0:
        from plot_curve import plot_map
        plot_map(val_map)

    def create_model(num_classes):
    # 注意，这里的backbone默认使用的是FrozenBatchNorm2d，即不会去更新bn参数
    # 目的是为了防止batch_size太小导致效果更差(如果显存很小，建议使用默认的FrozenBatchNorm2d)
    # 如果GPU显存很大可以设置比较大的batch_size就可以将norm_layer设置为普通的BatchNorm2d
    # trainable_layers包括['layer4', 'layer3', 'layer2', 'layer1', 'conv1']
    # 这里3表示只训练后三个阶段的参数['layer4', 'layer3', 'layer2']，5代表全部训练
    backbone = resnet50_fpn_backbone(norm_layer=torch.nn.BatchNorm2d,
                                     trainable_layers=3)
    # 训练自己数据集时不要修改这里的91，修改的是传入的num_classes参数
    # 因为要加载预训练模型，因此需要初始化指定类别数量的模型
    model = FasterRCNN(backbone=backbone, num_classes=91)

    # 载入预训练模型权重，权重下载地址:
    # https://download.pytorch.org/models/fasterrcnn_resnet50_fpn_coco-258fb6c6.pth
    weights_dict = torch.load("./backbone/fasterrcnn_resnet50_fpn_coco-258fb6c6.pth", map_location='cpu')
    # 加载预训练权重
    missing_keys, unexpected_keys = model.load_state_dict(weights_dict, strict=False)
    if len(missing_keys) != 0 or len(unexpected_keys) != 0:
        print("missing_keys: ", missing_keys)
        print("unexpected_keys: ", unexpected_keys)
    # 之后修改roi_head中的检测头，主要替换分类模块，替换为指定的类别参数
    # 得到分类模块输入的通道数量
    in_features = model.roi_heads.box_predictor.cls_score.in_features
    # 替换roi_head中的分类模块
    model.roi_heads.box_predictor = FastRCNNPredictor(in_features, num_classes)
    # 返回定义好的模型
    return model

    # 训练一个epoch
def train_one_epoch(model, optimizer, data_loader, device, epoch,
                    print_freq=50, warmup=False, scaler=None):
    # 模型改为训练模式
    model.train()
    # 日志
    metric_logger = utils.MetricLogger(delimiter="  ")
    metric_logger.add_meter('lr', utils.SmoothedValue(window_size=1, fmt='{value:.6f}'))
    header = 'Epoch: [{}]'.format(epoch)

    lr_scheduler = None
    # 当训练第一轮（epoch=0）时，启用warmup训练方式，可理解为热身训练
    if epoch == 0 and warmup is True:  
        warmup_factor = 1.0 / 1000
        warmup_iters = min(1000, len(data_loader) - 1)

        lr_scheduler = utils.warmup_lr_scheduler(optimizer, warmup_iters, warmup_factor)

    mloss = torch.zeros(1).to(device)  # mean losses
    for i, [images, targets] in enumerate(metric_logger.log_every(data_loader, print_freq, header)):
        # 将图片和标签放入指定的存储设备(cpu or gpu)，之后转化为列表格式
        images = list(image.to(device) for image in images)  
        targets = [{k: v.to(device) for k, v in t.items()} for t in targets]
        # 混合精度训练上下文管理器，如果在CPU环境中不起任何作用
        with torch.cuda.amp.autocast(enabled=scaler is not None):
            # 得到模型损失
            loss_dict = model(images, targets)
            # 所有模块的损失求和
            losses = sum(loss for loss in loss_dict.values())  

        # reduce losses over all GPUs for logging purpose
        # 多卡训练时这一行有用
        loss_dict_reduced = utils.reduce_dict(loss_dict)
        losses_reduced = sum(loss for loss in loss_dict_reduced.values())
        # 提取损失数值
        loss_value = losses_reduced.item()
        # 记录训练损失
        mloss = (mloss * i + loss_value) / (i + 1)  # update mean losses
        # 当计算的损失为无穷大时停止训练
        if not math.isfinite(loss_value):
            print("Loss is {}, stopping training".format(loss_value))
            print(loss_dict_reduced)
            sys.exit(1)

        optimizer.zero_grad()
        # 如果有梯度缩放操作的话，就走这一步
        if scaler is not None:
            scaler.scale(losses).backward()
            scaler.step(optimizer)
            scaler.update()
        else:
            # 反向传播，更新参数
            losses.backward()
            optimizer.step()
        # 第一轮使用warmup训练方式
        if lr_scheduler is not None:
            lr_scheduler.step()

        metric_logger.update(loss=losses_reduced, **loss_dict_reduced)
        now_lr = optimizer.param_groups[0]["lr"]
        metric_logger.update(lr=now_lr)
    # 返回平均损失和当前的学习率
    return mloss, now_lr
