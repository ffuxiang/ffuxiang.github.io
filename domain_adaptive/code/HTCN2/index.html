    class _fasterRCNN(nn.Module):
    """ faster RCNN """

    def __init__(self, classes, class_agnostic, lc, gc, la_attention=False, mid_attention=False):
        super(_fasterRCNN, self).__init__()
        self.classes = classes
        self.n_classes = len(classes)
        self.class_agnostic = class_agnostic
        # 损失
        self.RCNN_loss_cls = 0
        self.RCNN_loss_bbox = 0
        # 下面四个变量都默认True
        self.lc = lc
        self.gc = gc
        self.la_attention = la_attention
        self.mid_attention = mid_attention
        # RPN模块
        self.RCNN_rpn = _RPN(self.dout_base_model)
        self.RCNN_proposal_target = _ProposalTargetLayer(self.n_classes)
        # ROI模块
        self.RCNN_roi_pool = ROIPool((cfg.POOLING_SIZE, cfg.POOLING_SIZE), 1.0 / 16.0)
        self.RCNN_roi_align = ROIAlign((cfg.POOLING_SIZE, cfg.POOLING_SIZE), 1.0 / 16.0, 0)


    def forward(self, im_data, im_info, gt_boxes, num_boxes, target=False, eta=1.0):
        batch_size = im_data.size(0)

        im_info = im_info.data
        gt_boxes = gt_boxes.data
        num_boxes = num_boxes.data

        # 将图像数据传入浅层网络(对应论文G1)，提取浅层特征
        base_feat1 = self.RCNN_base1(im_data)
        if self.lc:
            # 将浅层特征传入像素点级的域分类器(对应论文D1)，逐像素点预测领域分数，返回一张域分类图
            # 注意在传入之前，需要先经过一层梯度反转层
            d_pixel, _ = self.netD_pixel(grad_reverse(base_feat1, lambd=eta))
            # if not target:
            # 这里得到浅层特征向量，主要是利用域分类器里面的两层卷积
            # 用于后续计算上下文向量，注意这里要切断梯度
            _, feat_pixel = self.netD_pixel(base_feat1.detach())
        else:
            d_pixel = self.netD_pixel(grad_reverse(base_feat1, lambd=eta))

        if self.la_attention:
            # 将预测到的领域分数图和特征图传入local_attention函数，得到经过局部特征掩模处理后的特征，即对浅层特征进行加权
            # 注意这里预测分数变量需要切断梯度，防止后续的操作影响域分类器的训练
            base_feat1 = local_attention(base_feat1, d_pixel.detach())
        # 再将处理后的特征传入中层网络(对应论文G2)，提取中层特征
        base_feat2 = self.RCNN_base2(base_feat1)
        if self.gc:
            # 将中层特征传入中层域分类器(对应论文D2)，预测领域分数，一共俩分数，分别表示该图属于两个领域的置信度
            domain_mid, _ = self.netD_mid(grad_reverse(base_feat2, lambd=eta))
            # if not target:
            # 这里和浅层特征处理方法一样，先切断梯度，然后得到中层特征向量，用于后续计算上下文向量
            _, feat_mid = self.netD_mid(base_feat2.detach())
        else:
            domain_mid = self.netD_mid(grad_reverse(base_feat2, lambd=eta))

        if self.mid_attention:
            # 将得到的领域分数和中层特征传入middle_attention函数，计算gi，即对中层特征进行加权，对应论文公式1、2
            # 这里领域预测分数同样也要切断梯度，用于保护域分类器的训练
            base_feat2 = middle_attention(base_feat2, domain_mid.detach())
        # 将中层特征传入深层网络，得到深层特征，即最终的特征图
        base_feat = self.RCNN_base3(base_feat2)
        if self.gc:
            # 将深层特征传入深层域分类器，对应论文D3，得到两个领域预测分数
            domain_p, _ = self.netD(grad_reverse(base_feat, lambd=eta))
            # 这里和前两次一样，得到深层特征，用于计算上下文向量
            _, feat = self.netD(base_feat.detach())
        else:
            domain_p = self.netD(grad_reverse(base_feat, lambd=eta))

        # feed base feature map tp RPN to obtain rois
        # 将特征传入RPN模块，得到感兴趣的区域
        rois, rpn_loss_cls, rpn_loss_bbox = self.RCNN_rpn(base_feat, im_info, gt_boxes, num_boxes)

        # if it is training phrase, then use ground trubut bboxes for refining
        if self.training:
            roi_data = self.RCNN_proposal_target(rois, gt_boxes, num_boxes)
            rois, rois_label, rois_target, rois_inside_ws, rois_outside_ws = roi_data

            rois_label = Variable(rois_label.view(-1).long())
            rois_target = Variable(rois_target.view(-1, rois_target.size(2)))
            rois_inside_ws = Variable(rois_inside_ws.view(-1, rois_inside_ws.size(2)))
            rois_outside_ws = Variable(rois_outside_ws.view(-1, rois_outside_ws.size(2)))
        else:
            rois_label = None
            rois_target = None
            rois_inside_ws = None
            rois_outside_ws = None
            rpn_loss_cls = 0
            rpn_loss_bbox = 0

        rois = Variable(rois)
        # do roi pooling based on predicted rois

        if cfg.POOLING_MODE == 'align':
            pooled_feat = self.RCNN_roi_align(base_feat, rois.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            pooled_feat = self.RCNN_roi_pool(base_feat, rois.view(-1, 5))

        # feed pooled features to top model
        pooled_feat = self._head_to_tail(pooled_feat)

        feat_pixel = feat_pixel.view(1, -1).repeat(pooled_feat.size(0), 1)
        feat_mid = feat_mid.view(1, -1).repeat(pooled_feat.size(0), 1)
        feat = feat.view(1, -1).repeat(pooled_feat.size(0), 1)
        # 先将前三组特征向量合并
        feat = torch.cat((feat_mid, feat), 1)
        feat = torch.cat((feat_pixel, feat), 1)
        # 将合并向量和实例区域向量传入RandomLayer，执行随机化操作，对应论文公式(5)
        # 此时传入的feat长度为384(128*3),pooled_feat为2048(以resnet为例)
        feat_random = self.RandomLayer([pooled_feat, feat])
        # 实例级域分类器，对齐每个实例的特征，对应论文公式(6)
        d_ins = self.netD_da(grad_reverse(feat_random, lambd=eta))
        # 如果是目标域图像的话，则直接返回四个域分类器结果就行，只计算对抗损失
        # 否则需要继续往下走，进一步计算边界框回归参数，计算目标检测损失
        if target:
            return d_pixel, domain_p, domain_mid, d_ins
        # 直接对特征进行拼接，这里参考算法Strong-Weak Distribution Alignment，利用了上下文特征来增强检测器的性能
        pooled_feat = torch.cat((feat, pooled_feat), 1)
        # 之后计算边界框回归参数
        bbox_pred = self.RCNN_bbox_pred(pooled_feat)
        # 计算边界框回归损失
        if self.training and not self.class_agnostic:
            bbox_pred_view = bbox_pred.view(bbox_pred.size(0), int(bbox_pred.size(1) / 4), 4)
            bbox_pred_select = torch.gather(bbox_pred_view, 1,
                                            rois_label.view(rois_label.size(0), 1, 1).expand(rois_label.size(0), 1, 4))
            bbox_pred = bbox_pred_select.squeeze(1)

        # 计算类别概率
        cls_score = self.RCNN_cls_score(pooled_feat)
        cls_prob = F.softmax(cls_score, 1)

        RCNN_loss_cls = 0
        RCNN_loss_bbox = 0
        # 计算分类损失
        if self.training:
            # classification loss
            RCNN_loss_cls = F.cross_entropy(cls_score, rois_label)

            # bounding box regression L1 loss
            RCNN_loss_bbox = _smooth_l1_loss(bbox_pred, rois_target, rois_inside_ws, rois_outside_ws)

        cls_prob = cls_prob.view(batch_size, rois.size(1), -1)
        bbox_pred = bbox_pred.view(batch_size, rois.size(1), -1)
        # 最后返回相关变量
        return rois, cls_prob, bbox_pred, rpn_loss_cls, rpn_loss_bbox, RCNN_loss_cls, RCNN_loss_bbox, rois_label, d_pixel, domain_p, domain_mid, d_ins  # ,diff

    class GradReverse(Function):
    def __init__(self, lambd):
        self.lambd = lambd

    def forward(self, x):
        # 前向传播不做处理
        return x.view_as(x)

    def backward(self, grad_output):
        # 只修改反向传播，多乘以-1
        return (grad_output * -self.lambd)

def grad_reverse(x, lambd=1.0):
    return GradReverse(lambd)(x)

    像素级域分类器

    对应论文D_1

    # 对浅层特征执行领域分类的域分类器，像素级别，按像素点对齐
# 这里与SW里的函数类似
class netD_pixel(nn.Module):
    def __init__(self, context=False):
        super(netD_pixel, self).__init__()
        # 定义好卷积层，通道数依次降低，注意，这里卷积核均为1*1
        self.conv1 = nn.Conv2d(256, 256, kernel_size=1, stride=1,
                               padding=0, bias=False)
        self.conv2 = nn.Conv2d(256, 128, kernel_size=1, stride=1,
                               padding=0, bias=False)
        # 最后一层通道数变为1
        self.conv3 = nn.Conv2d(128, 1, kernel_size=1, stride=1,
                               padding=0, bias=False)
        # 判断后续是否需要计算上下文向量，默认True
        self.context = context
        #初始化参数
        self._init_weights()

    def _init_weights(self):
        def normal_init(m, mean, stddev, truncated=False):
            """
            weight initalizer: truncated normal and random normal.
            """
            if truncated:
                m.weight.data.normal_().fmod_(2).mul_(stddev).add_(mean)  # not a perfect approximation
            else:
                m.weight.data.normal_(mean, stddev)

        normal_init(self.conv1, 0, 0.01)
        normal_init(self.conv2, 0, 0.01)
        normal_init(self.conv3, 0, 0.01)

    def forward(self, x):
        # 依次经过conv1->relu->conv2->relu
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2(x))
        if self.context:
            # 如果用到上下文的话，则还需要返回浅层特征向量
            # 向量长度为128
            feat = F.avg_pool2d(x, (x.size(2), x.size(3)))
            # 经过最后一层卷积，得到预测图
            x = self.conv3(x)
            # 返回预测分数图和特征向量
            return F.sigmoid(x), feat
        else:
            x = self.conv3(x)
            return F.sigmoid(x)

    # 对中层特征进行领域分类的域分类器，输出两个值
# 和SW里面的全局特征对齐模块类似
class netD_mid(nn.Module):
    def __init__(self, context=False):
        super(netD_mid, self).__init__()
        # 定义好卷积，标准化层，线性回归层
        # 与像素级域分类器的区别在于，这里卷积核尺寸为3*3，并且步长为2，填充为1
        # 并且最后经过线性回归，返回两个值
        self.conv1 = conv3x3(512, 512, stride=2)
        self.bn1 = nn.BatchNorm2d(512)
        self.conv2 = conv3x3(512, 128, stride=2)
        self.bn2 = nn.BatchNorm2d(128)
        self.conv3 = conv3x3(128, 128, stride=2)
        self.bn3 = nn.BatchNorm2d(128)
        self.fc = nn.Linear(128, 2)
        self.context = context

    def forward(self, x):
        # 按顺序经过卷积、标准化、relu、dropout
        x = F.dropout(F.relu(self.bn1(self.conv1(x))), training=self.training)
        x = F.dropout(F.relu(self.bn2(self.conv2(x))), training=self.training)
        x = F.dropout(F.relu(self.bn3(self.conv3(x))), training=self.training)
        # 拉伸成向量，便于后续传入线性回归
        x = F.avg_pool2d(x, (x.size(2), x.size(3)))
        x = x.view(-1, 128)
        if self.context:
            # 保存向量，用于后续计算上下文特征，向量长度为128
            feat = x
        x = self.fc(x)
        if self.context:
            return x, feat
        else:
            return x

class netD(nn.Module):
    def __init__(self, context=False):

        self.bn1 = nn.BatchNorm2d(512)

        x = x.view(-1, 128)
        if self.context:

            feat = x
        x = self.fc(x)
        if self.context:
            return x, feat
        else:
            return x

class netD_da(nn.Module):
    def __init__(self, feat_d):

        # 最后输出两个值
        self.fc1 = nn.Linear(feat_d, 100)

        self.bn2 = nn.BatchNorm1d(100)

        x = F.dropout(F.relu(self.bn1(self.fc1(x))), training=self.training)
        x = F.dropout(F.relu(self.bn2(self.fc2(x))), training=self.training)
        x = self.fc3(x)
        # 返回预测结果
        return x  # [256, 2]，第一个维度表示实例个数

    特征加权模块¶

    浅层特征加权

    对应论文里的局部特征掩模模块

    def local_attention(features, d):
    # features.size() =  [1, 256, h, w] 特征图尺寸
    # d.size() = [1, 1, h, w]  浅层特征的领域预测分数
    # 限制d大小(1e-6到1之间)，防止出现零的情况
    d = d.clamp(1e-6, 1)
    # 计算信息熵H，对应论文公式1
    H = - (d * d.log() + (1 - d) * (1 - d).log())
    # 计算权重
    w = 1 - H
    # 对浅层特征图进行加权，对应论文公式2
    features_new = (1 + w) * features

    return features_new

    def middle_attention(features, d):
    # d.size() = [1,2]
    # 求信息熵
    _, H = self_entropy(d, softmax=True)
    # 对中层特征进行加权
    features_new = (1 + H) * features

    return features_new

    计算信息熵

    def self_entropy(prob, softmax):
    # prob.size() = [N, C(2)]   C: Number of categories
    # 由于域分类器是直接输出预测分数，没有经过归一化
    # 因此这里首先需要经过softmax归一化
    if softmax:
        prob = F.softmax(prob, 1)
    # 限制预测分数的大小，防止出现0
    prob = prob.clamp(1e-6, 1)
    log_prob = torch.log(prob)
    # 计算信息熵
    H = - torch.sum(prob * log_prob, dim=1)  # [N]
    # 为什么要计算均值？对同一个batch里图像的信息熵求均值？
    H_mean = H.mean()

    return H, H_mean

    随机化操作¶

    # 主干网络以resnet为例
class RandomLayer(nn.Module):
    def __init__(self, input_dim_list=[], output_dim=1024):
        super(RandomLayer, self).__init__()
        # input_dim_list默认[2048,384]
        self.input_num = len(input_dim_list)  # 2
        # output_dim默认1024
        self.output_dim = output_dim
        # 生成两个随机矩阵，对应论文中R1:[384,1024],R2:[2048,1024]
        self.random_matrix = [torch.rand(input_dim_list[i], output_dim) for i in range(self.input_num)]

    def forward(self, input_list):
        # 依次做矩阵乘积fc*R1->[batch,1024],f_{ins}*R2->[batch,1024]
        return_list = [torch.mm(input_list[i], self.random_matrix[i]) for i in range(self.input_num)]
        # 这里表示两个数做除法，分母为分之根d，分子为公式(5)的第一项
        return_tensor = return_list[0] / math.pow(float(self.output_dim), 1.0 / len(return_list))
        for single in return_list[1:]:
            # 然后再遍历做点乘，但这里只遍历一次
            return_tensor = torch.mul(return_tensor, single)
        # 其实完全可以先做点乘，然后再除以根d
        # 返回随机处理后的特征
        return return_tensor

    def cuda(self):
        super(RandomLayer, self).cuda()
        # 将数据传入显卡中
        self.random_matrix = [val.cuda() for val in self.random_matrix]
