    class _fasterRCNN(nn.Module):
    """ faster RCNN """
    def __init__(self, classes, class_agnostic,lc,gc):
        super(_fasterRCNN, self).__init__()
        self.classes = classes
        self.n_classes = len(classes)
        self.class_agnostic = class_agnostic
        # loss
        self.RCNN_loss_cls = 0
        self.RCNN_loss_bbox = 0
        self.lc = lc
        self.gc = gc
        # define rpn
        self.RCNN_rpn = _RPN(self.dout_base_model)
        self.RCNN_proposal_target = _ProposalTargetLayer(self.n_classes)
        self.RCNN_roi_pool = _RoIPooling(cfg.POOLING_SIZE, cfg.POOLING_SIZE, 1.0/16.0)
        self.RCNN_roi_align = RoIAlignAvg(cfg.POOLING_SIZE, cfg.POOLING_SIZE, 1.0/16.0)

        self.grid_size = cfg.POOLING_SIZE * 2 if cfg.CROP_RESIZE_WITH_MAX_POOL else cfg.POOLING_SIZE
        self.RCNN_roi_crop = _RoICrop()

    def forward(self, im_data, im_info, gt_boxes, num_boxes, target=False, eta=1.0):
        batch_size = im_data.size(0)

        im_info = im_info.data
        gt_boxes = gt_boxes.data
        num_boxes = num_boxes.data

        # feed image data to base model to obtain base feature map
        # 数据传入特征提取网络前半段，得到浅层特征
        base_feat1 = self.RCNN_base1(im_data)
        # self.netD_pixel与self.netD都在resnet_global_local.py文件里
        # 分别对应浅层特征对齐和深层特征对齐
        # self.lc表示加入了初级对齐模块后，是否添加上下文正则化项
        if self.lc:
            # 特征在对齐之前要先套一个梯度反转模块，即这里的grad_reverse
            # d_pixel表示一张浅层特征对齐图(对特征图上元素领域的预测)，通道数为1
            # 即每个位置只生成一个元素，趋向于1表示目标域，趋向于0表示源域
            d_pixel, _ = self.netD_pixel(grad_reverse(base_feat1, lambd=eta))
            # print(d_pixel)
            if not target:
                # 如果该数据不是目标域的数据，则再得到特征向量，这里向量长度为128(经过了两层卷积以及一层全局平均池化)
                # 为什么要切断梯度？
                _, feat_pixel = self.netD_pixel(base_feat1.detach())
        else:
            # 如果不加上下文正则化项的话，则只得特征对齐图即可
            d_pixel = self.netD_pixel(grad_reverse(base_feat1, lambd=eta))
        # 浅层特征传入特征提取网络后半段，得到深层特征
        base_feat = self.RCNN_base2(base_feat1)
        # 是否添加上下文正则化器
        if self.gc:
            # domain_p为两个数，两个数分别表示图像是源域的置信度和目标域的置信度
            # 这里与浅层特征对齐不同，这里是俩数，浅层特征对齐返回一张图
            domain_p, _ = self.netD(grad_reverse(base_feat, lambd=eta))
            if target:
                # 如果是目标域图像，则只返回两个特征对齐图，用于计算域分类损失
                # 目标域图像无坐标，因此无法计算faster r-cnn损失，只能计算域分类损失
                return d_pixel, domain_p
            # 再得到特征向量，这里向量长度也为128
            _, feat = self.netD(base_feat.detach())
        else:
            # 如果不加上下文正则化器的话，则无需计算特征向量
            domain_p = self.netD(grad_reverse(base_feat, lambd=eta))
            if target:
                return d_pixel,domain_p
        # feed base feature map tp RPN to obtain rois
        # RPN模块
        rois, rpn_loss_cls, rpn_loss_bbox = self.RCNN_rpn(base_feat, im_info, gt_boxes, num_boxes)

        # if it is training phrase, then use ground trubut bboxes for refining
        if self.training:
            roi_data = self.RCNN_proposal_target(rois, gt_boxes, num_boxes)
            rois, rois_label, rois_target, rois_inside_ws, rois_outside_ws = roi_data

            rois_label = Variable(rois_label.view(-1).long())
            rois_target = Variable(rois_target.view(-1, rois_target.size(2)))
            rois_inside_ws = Variable(rois_inside_ws.view(-1, rois_inside_ws.size(2)))
            rois_outside_ws = Variable(rois_outside_ws.view(-1, rois_outside_ws.size(2)))
        else:
            rois_label = None
            rois_target = None
            rois_inside_ws = None
            rois_outside_ws = None
            rpn_loss_cls = 0
            rpn_loss_bbox = 0

        rois = Variable(rois)
        # do roi pooling based on predicted rois
        # ROI模块
        if cfg.POOLING_MODE == 'crop':
            # pdb.set_trace()
            # pooled_feat_anchor = _crop_pool_layer(base_feat, rois.view(-1, 5))
            grid_xy = _affine_grid_gen(rois.view(-1, 5), base_feat.size()[2:], self.grid_size)
            grid_yx = torch.stack([grid_xy.data[:,:,:,1], grid_xy.data[:,:,:,0]], 3).contiguous()
            pooled_feat = self.RCNN_roi_crop(base_feat, Variable(grid_yx).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat = F.max_pool2d(pooled_feat, 2, 2)
        elif cfg.POOLING_MODE == 'align':
            pooled_feat = self.RCNN_roi_align(base_feat, rois.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            pooled_feat = self.RCNN_roi_pool(base_feat, rois.view(-1,5))

        # feed pooled features to top model
        pooled_feat = self._head_to_tail(pooled_feat)
        # feat_pixel = torch.zeros(feat_pixel.size()).cuda()
        # 如果使用上下文正则化器的话，则需要将三个特征向量进行拼接
        if self.lc:
            feat_pixel = feat_pixel.view(1, -1).repeat(pooled_feat.size(0), 1)
            pooled_feat = torch.cat((feat_pixel, pooled_feat), 1)
        if self.gc:
            feat = feat.view(1, -1).repeat(pooled_feat.size(0), 1)
            pooled_feat = torch.cat((feat, pooled_feat), 1)
            # compute bbox offset

        # compute bbox offset
        # 预测回归参数
        bbox_pred = self.RCNN_bbox_pred(pooled_feat)
        if self.training and not self.class_agnostic:
            bbox_pred_view = bbox_pred.view(bbox_pred.size(0), int(bbox_pred.size(1) / 4), 4)
            bbox_pred_select = torch.gather(bbox_pred_view, 1, rois_label.view(rois_label.size(0), 1, 1).expand(rois_label.size(0), 1, 4))
            bbox_pred = bbox_pred_select.squeeze(1)

        # compute object classification probability
        # 预测类别
        cls_score = self.RCNN_cls_score(pooled_feat)
        cls_prob = F.softmax(cls_score, 1)

        RCNN_loss_cls = 0
        RCNN_loss_bbox = 0

        if self.training:
            # 计算faster rcnn损失
            # classification loss
            RCNN_loss_cls = F.cross_entropy(cls_score, rois_label)
            # bounding box regression L1 loss
            RCNN_loss_bbox = _smooth_l1_loss(bbox_pred, rois_target, rois_inside_ws, rois_outside_ws)

        cls_prob = cls_prob.view(batch_size, rois.size(1), -1)
        bbox_pred = bbox_pred.view(batch_size, rois.size(1), -1)

        return rois, cls_prob, bbox_pred, rpn_loss_cls, rpn_loss_bbox, RCNN_loss_cls, RCNN_loss_bbox, rois_label,d_pixel, domain_p

    梯度反转模块¶

    class GradReverse(Function):
    def __init__(self, lambd):
        self.lambd = lambd

    def forward(self, x):
        return x.view_as(x)

    def backward(self, grad_output):
        #pdb.set_trace()
        # 当反向传播传播到这里的时候，梯度取负数
        return (grad_output * -self.lambd) 


def grad_reverse(x, lambd=1.0):
    return GradReverse(lambd)(x)

    class netD_pixel(nn.Module):

    def __init__(self, context=False):
        super(netD_pixel, self).__init__()
        # 定义三层卷积层，这里卷积层核均为1，与深层特征对齐不同
        self.conv1 = nn.Conv2d(256, 256, kernel_size=1, stride=1,
                  padding=0, bias=False)
        self.conv2 = nn.Conv2d(256, 128, kernel_size=1, stride=1,
                               padding=0, bias=False)
        self.conv3 = nn.Conv2d(128, 1, kernel_size=1, stride=1,
                               padding=0, bias=False)
        # 是否加入上下文正则化器
        self.context = context
        # 初始化参数
        self._init_weights()

    def _init_weights(self):

      def normal_init(m, mean, stddev, truncated=False):
        """
        weight initalizer: truncated normal and random normal.
        """
        # x is a parameter
        if truncated:
          m.weight.data.normal_().fmod_(2).mul_(stddev).add_(mean)  # not a perfect approximation
        else:
          m.weight.data.normal_(mean, stddev)
          #m.bias.data.zero_()

      normal_init(self.conv1, 0, 0.01)
      normal_init(self.conv2, 0, 0.01)
      normal_init(self.conv3, 0, 0.01)

    def forward(self, x):
        # 前向传播
        x = F.relu(self.conv1(x))
        x = F.relu(self.conv2(x))
        if self.context:
          # 相当于一个全局平均池化操作，将特征图变为特征向量
          feat = F.avg_pool2d(x, (x.size(2), x.size(3)))
          # 特征图再经过conv3，变为一层图片，称为特征对齐图
          x = self.conv3(x)
          # 特征对齐图经过sigmoid归一化
          return F.sigmoid(x),feat
        else:
          # 如果不引入上下文正则化器的话，则只返回特征对齐图
          x = self.conv3(x)
          return F.sigmoid(x)

    class netD(nn.Module):
    def __init__(self,context=False):
        super(netD, self).__init__()
        # 注意conv3x3为卷积核尺寸为3，填充为1的卷积操作
        # 这里定义了三层卷积，卷积步幅设置为2
        # 并且依次定义三层标准化层
        self.conv1 = conv3x3(1024, 512, stride=2)
        self.bn1 = nn.BatchNorm2d(512)
        self.conv2 = conv3x3(512, 128, stride=2)
        self.bn2 = nn.BatchNorm2d(128)
        self.conv3 = conv3x3(128, 128, stride=2)
        self.bn3 = nn.BatchNorm2d(128)
        # 定义一层全连接层，回归出两个数，分别代表该图是源域的置信度还是目标域的置信度
        self.fc = nn.Linear(128,2)
        self.context = context
        self.leaky_relu = nn.LeakyReLU(negative_slope=0.2, inplace=True)

    def forward(self, x):
        # 依次经过卷积、标准化、relu、dropout
        x = F.dropout(F.relu(self.bn1(self.conv1(x))),training=self.training)
        x = F.dropout(F.relu(self.bn2(self.conv2(x))),training=self.training)
        x = F.dropout(F.relu(self.bn3(self.conv3(x))),training=self.training)
        # 经过全局平均池化，方便后续传入全连接层
        x = F.avg_pool2d(x, (x.size(2),x.size(3)))
        x = x.view(-1,128)
        if self.context:
          # 如果添加上下文正则化器的话，则保存此时的特征向量
          feat = x
        x = self.fc(x)
        if self.context:
          return x,feat
        else:
          return x

    FL损失¶

    class FocalLoss(nn.Module):
    r"""
        This criterion is a implemenation of Focal Loss, which is proposed in
        Focal Loss for Dense Object Detection.

            Loss(x, class) = - \alpha (1-softmax(x)[class])^gamma \log(softmax(x)[class])

        The losses are averaged across observations for each minibatch.
        Args:
            alpha(1D Tensor, Variable) : the scalar factor for this criterion
            gamma(float, double) : gamma > 0; reduces the relative loss for well-classiﬁed examples (p > .5),
                                   putting more focus on hard, misclassiﬁed examples
            size_average(bool): size_average(bool): By default, the losses are averaged over observations for each minibatch.
                                However, if the field size_average is set to False, the losses are
                                instead summed for each minibatch.
    """

    def __init__(self, class_num, alpha=None, gamma=2, size_average=True, sigmoid=False, reduce=True):
        super(FocalLoss, self).__init__()
        if alpha is None:
            # alpha默认设置为1
            self.alpha = Variable(torch.ones(class_num, 1) * 1.0)
        else:
            if isinstance(alpha, Variable):
                self.alpha = alpha
            else:
                self.alpha = Variable(alpha)
        # self.gamma表示FL函数中的γ，即指数
        self.gamma = gamma
        self.class_num = class_num
        # 是否取均值
        self.size_average = size_average
        # 归一化策略
        self.sigmoid = sigmoid
        self.reduce = reduce

    def forward(self, inputs, targets):
        # batch数量
        N = inputs.size(0)
        C = inputs.size(1)
        if self.sigmoid:
            # 将输出的置信度经过sigmoid归一化操作
            P = F.sigmoid(inputs)
            # 如果是源域数据
            if targets == 0:
                # 源域数据p_t=1-p，这里probs表示论文中的p_t
                probs = 1 - P
                log_p = probs.log()
                # 计算源域中的FL损失，对应论文公式(5)
                batch_loss = - (torch.pow((1 - probs), self.gamma)) * log_p
            # 如果是目标域数据
            if targets == 1:
                # 目标域中，p_t=p
                probs = P
                log_p = probs.log()
                # 计算目标域中的FL损失，对应论文公式(6)
                batch_loss = - (torch.pow((1 - probs), self.gamma)) * log_p
        else:
            # softmax归一化策略，经过softmax归一化之后，两个值和为1，真正意义上变为概率
            # 因此p_t等于对应位置上的数值，省去了1-p这一步骤
            P = F.softmax(inputs)
            # 创建一个与inputs尺寸相同的全零数组
            class_mask = inputs.data.new(N, C).fill_(0)
            # 变为张量，可导
            class_mask = Variable(class_mask)
            # 标签拉伸
            ids = targets.view(-1, 1)
            # 沿第一维度对变量class_mask索引，将标签位置的元素设置为1
            # 如果是源域图像，标签为0，则将第一列元素设置为1
            class_mask.scatter_(1, ids.data, 1.)

            if inputs.is_cuda and not self.alpha.is_cuda:
                self.alpha = self.alpha.cuda()
            alpha = self.alpha[ids.data.view(-1)]
            # P * class_mask表示提取期望的预测值，即如果标签为源域，则提取源域预测的置信度
            # 之后在第一维度求和，拉伸，得到p_t
            # 如果只预测了一种深层特征，则可省略求和这一步
            probs = (P * class_mask).sum(1).view(-1, 1)
            # 求对数
            log_p = probs.log()
            # 求FL损失，对应论文公式(4)
            batch_loss = -alpha * (torch.pow((1 - probs), self.gamma)) * log_p

        if not self.reduce:
            return batch_loss
        # 返回平均损失或求和损失
        if self.size_average:
            loss = batch_loss.mean()
        else:
            loss = batch_loss.sum()
        return loss
