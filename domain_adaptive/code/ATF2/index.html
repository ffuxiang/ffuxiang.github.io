    class _fasterRCNN(nn.Module):
    """ faster RCNN """

    def __init__(self, classes, class_agnostic):
        super(_fasterRCNN, self).__init__()
        self.classes = classes
        self.n_classes = len(classes)
        self.class_agnostic = class_agnostic
        # loss
        self.RCNN_loss_cls = 0
        self.RCNN_loss_bbox = 0

        # define rpn
        # 定义RPN模块
        self.RCNN_rpn = _RPN(self.dout_base_model)
        # 这个模块后面没用到，应该是多定义的
        self.RCNN_rpn_t = _RPN(self.dout_base_model)
        # 筛选ROI Head中的proposals，只在源域的训练阶段执行
        self.RCNN_proposal_target = _ProposalTargetLayer(self.n_classes)
        # 定义三种ROI Pooling操作
        self.RCNN_roi_pool = _RoIPooling(cfg.POOLING_SIZE, cfg.POOLING_SIZE, 1.0 / 16.0)
        self.RCNN_roi_align = RoIAlignAvg(cfg.POOLING_SIZE, cfg.POOLING_SIZE, 1.0 / 16.0)
        self.grid_size = cfg.POOLING_SIZE * 2 if cfg.CROP_RESIZE_WITH_MAX_POOL else cfg.POOLING_SIZE
        self.RCNN_roi_crop = _RoICrop()
        # 定义三个阶段的图像级域分类器
        self.RCNN_imageDA_3 = _ImageDA(256)
        self.RCNN_imageDA_4 = _ImageDA(512)
        self.RCNN_imageDA = _ImageDA(self.dout_base_model)
        # 定义实例级域分类器
        self.RCNN_instanceDA = _InstanceDA()

      def _init_modules(self):
    resnet = resnet101()

    if self.pretrained == True:
      print("Loading pretrained weights from %s" %(self.model_path))
      state_dict = torch.load(self.model_path)
      resnet.load_state_dict({k:v for k,v in state_dict.items() if k in resnet.state_dict()})

    # Build resnet.
    self.conv3 = nn.Sequential(resnet.conv1, resnet.bn1,resnet.relu,
      resnet.maxpool,resnet.layer1)
    self.conv34 = nn.Sequential(resnet.layer2)
    self.conv45 = nn.Sequential(resnet.layer3)
    self.RCNN_base = nn.Sequential(resnet.conv1, resnet.bn1,resnet.relu,
      resnet.maxpool,resnet.layer1,resnet.layer2,resnet.layer3)

    self.RCNN_top = nn.Sequential(resnet.layer4)

    self.RCNN_cls_score = nn.Linear(2048, self.n_classes)
    if self.class_agnostic:
      self.RCNN_bbox_pred = nn.Linear(2048, 4)
    else:
      self.RCNN_bbox_pred = nn.Linear(2048, 4 * self.n_classes)

    # Fix blocks
    for p in self.RCNN_base[0].parameters(): p.requires_grad=False
    for p in self.RCNN_base[1].parameters(): p.requires_grad=False

    assert (0 <= cfg.RESNET.FIXED_BLOCKS < 4)
    if cfg.RESNET.FIXED_BLOCKS >= 3:
      for p in self.RCNN_base[6].parameters(): p.requires_grad=False
    if cfg.RESNET.FIXED_BLOCKS >= 2:
      for p in self.RCNN_base[5].parameters(): p.requires_grad=False
    if cfg.RESNET.FIXED_BLOCKS >= 1:
      for p in self.RCNN_base[4].parameters(): p.requires_grad=False

    def set_bn_fix(m):
      classname = m.__class__.__name__
      if classname.find('BatchNorm') != -1:
        for p in m.parameters(): p.requires_grad=False

    self.RCNN_base.apply(set_bn_fix)
    self.RCNN_top.apply(set_bn_fix)

    前向传播¶

       def forward(self, im_data, im_info, gt_boxes, num_boxes, need_backprop,
                tgt_im_data, tgt_im_info, tgt_gt_boxes, tgt_num_boxes, tgt_need_backprop):
        """
        im_data: 一张源域图像
        im_info: 源域图片长宽信息
        gt_boxes: 源域图像标签边界框
        num_boxes: 源域图像标签框数量
        need_backprop: 领域标签，源域数据对应设为1
        tgt_im_data: 一张目标域图像
        tgt_im_info: 目标域图片长宽信息
        tgt_gt_boxes: 目标域边界框标签 (此变量是随机初始化的，因为目标域无标签)
        tgt_num_boxes: 目标域标签框数量 (同上，随机初始化的数据)
        tgt_need_backprop:领域标签，目标域数据对应设为0
        """

        assert need_backprop.detach() == 1 and tgt_need_backprop.detach() == 0
        # 首先处理源域数据
        # 分别得到输入数据的信息
        batch_size = im_data.size(0)
        im_info = im_info.data  # (size1,size2, image ratio(new image / source image) )
        gt_boxes = gt_boxes.data
        num_boxes = num_boxes.data

        need_backprop = need_backprop.data

        # feed image data to base model to obtain base feature map
        # 将图片依次传入特征提取层，得到不同阶段的特征图
        # 其中，后缀是s的表示Chief Net
        conv3_feat = self.conv3_s(im_data)
        conv4_feat = self.conv34_s(conv3_feat)
        base_feat = self.conv45_s(conv4_feat)
        # 后缀是t的表示Ancillary Net
        conv3_feat_t = self.conv3_t(im_data)
        conv4_feat_t = self.conv34_t(conv3_feat_t)
        base_feat_t = self.conv45_t(conv4_feat_t)

        # feed base feature map tp RPN to obtain rois
        # 将特征图依次传入RPN中，得到proposals(区域建议框)和rpn loss(rpn损失)
        # 这里得到的proposals是经过RPN模块中筛选后的proposals
        # 即先根据预测分数筛选前XX个，之后经过NMS筛选，最后再根据预测分数筛选前XX个，和后面ROI Head的筛选不同
        self.RCNN_rpn.train()
        rois_domain, rpn_loss_cls1, rpn_loss_bbox1, _ = self.RCNN_rpn(base_feat, im_info, gt_boxes, num_boxes)
        rois_domain_t, rpn_loss_cls2, rpn_loss_bbox2, _ = self.RCNN_rpn(base_feat_t, im_info, gt_boxes, num_boxes)
        # 两种损失分别相加，得到rpn最终的分类损失和边界框回归损失
        rpn_loss_cls = rpn_loss_cls1 + rpn_loss_cls2
        rpn_loss_bbox = rpn_loss_bbox1 + rpn_loss_bbox2

        # if it is training phrase, then use ground trubut bboxes for refining
        # 如果是训练阶段，则需要对每个proposal打上标签，并且筛选特定正负样本比例的proposals传入roi head模块
        # 这里的筛选操作是ROI Head中的筛选(筛选特定正负样本，用于计算roi损失)，和上面RPN中的筛选不同(一共会经过两次筛选)
        if self.training:
            # 得到筛选后的proposals
            roi_data = self.RCNN_proposal_target(rois_domain, gt_boxes, num_boxes)
            rois, rois_label, rois_target, rois_inside_ws, rois_outside_ws = roi_data
            # 使数据变为张量格式->可导
            rois_label = Variable(rois_label.view(-1).long())
            rois_target = Variable(rois_target.view(-1, rois_target.size(2)))
            rois_inside_ws = Variable(rois_inside_ws.view(-1, rois_inside_ws.size(2)))
            rois_outside_ws = Variable(rois_outside_ws.view(-1, rois_outside_ws.size(2)))
            # 对Ancillary Net中的proposals执行同样的操作
            roi_data_t = self.RCNN_proposal_target(rois_domain_t, gt_boxes, num_boxes)
            rois_t, rois_label_t, rois_target_t, rois_inside_ws_t, rois_outside_ws_t = roi_data_t

            rois_label_t = Variable(rois_label_t.view(-1).long())
            rois_target_t = Variable(rois_target_t.view(-1, rois_target_t.size(2)))
            rois_inside_ws_t = Variable(rois_inside_ws_t.view(-1, rois_inside_ws_t.size(2)))
            rois_outside_ws_t = Variable(rois_outside_ws_t.view(-1, rois_outside_ws_t.size(2)))

        else:
            # 测试阶段的话不进行筛选、打标签，将所有的proposals都传入roi head
            rois_label = None
            rois_target = None
            rois_inside_ws = None
            rois_outside_ws = None
            rpn_loss_cls = 0
            rpn_loss_bbox = 0

        # 选择不同的roi pooling操作，得到pooling features(区域特征)
        # 下面这一块只对筛选后的proposals执行roi pooling操作，用于计算roi回归损失和分类损失
        if cfg.POOLING_MODE == 'crop':
            grid_xy = _affine_grid_gen(rois.view(-1, 5), base_feat.size()[2:], self.grid_size)
            grid_yx = torch.stack([grid_xy.data[:, :, :, 1], grid_xy.data[:, :, :, 0]], 3).contiguous()
            pooled_feat = self.RCNN_roi_crop(base_feat, Variable(grid_yx).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat = F.max_pool2d(pooled_feat, 2, 2)

            grid_xy_t = _affine_grid_gen(rois_t.view(-1, 5), base_feat_t.size()[2:], self.grid_size)
            grid_yx_t = torch.stack([grid_xy_t.data[:, :, :, 1], grid_xy_t.data[:, :, :, 0]], 3).contiguous()
            pooled_feat_t = self.RCNN_roi_crop(base_feat_t, Variable(grid_yx_t).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat_t = F.max_pool2d(pooled_feat_t, 2, 2)

        elif cfg.POOLING_MODE == 'align':
            pooled_feat = self.RCNN_roi_align(base_feat, rois.view(-1, 5))
            pooled_feat_t = self.RCNN_roi_align(base_feat_t, rois_t.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            pooled_feat = self.RCNN_roi_pool(base_feat, rois.view(-1, 5))
            pooled_feat_t = self.RCNN_roi_pool(base_feat_t, rois_t.view(-1, 5))

        # 下面对rpn模块得到的所有proposals执行roi pooling操作，用于计算实例级特征对齐损失
        # 这里的proposals是经过RPN模块内部对锚点的筛选得到的区域
        # 注意，后面只用到了辅助网络的proposals，主网中的源域分支不参与域适应损失的计算
        if cfg.POOLING_MODE == 'crop':
            grid_xy = _affine_grid_gen(rois_domain.view(-1, 5), base_feat.size()[2:], self.grid_size)
            grid_yx = torch.stack([grid_xy.data[:, :, :, 1], grid_xy.data[:, :, :, 0]], 3).contiguous()
            pooled_feat_domain = self.RCNN_roi_crop(base_feat, Variable(grid_yx).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat_domain = F.max_pool2d(pooled_feat, 2, 2)

            grid_xy_t = _affine_grid_gen(rois_domain_t.view(-1, 5), base_feat_t.size()[2:], self.grid_size)
            grid_yx_t = torch.stack([grid_xy_t.data[:, :, :, 1], grid_xy_t.data[:, :, :, 0]], 3).contiguous()
            pooled_feat_domain_t = self.RCNN_roi_crop(base_feat_t, Variable(grid_yx_t).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat_domain_t = F.max_pool2d(pooled_feat_t, 2, 2)
        elif cfg.POOLING_MODE == 'align':
            pooled_feat_domain = self.RCNN_roi_align(base_feat, rois_domain.view(-1, 5))
            pooled_feat_domain_t = self.RCNN_roi_align(base_feat_t, rois_domain_t.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            pooled_feat_domain = self.RCNN_roi_pool(base_feat, rois_domain.view(-1, 5))
            pooled_feat_domain_t = self.RCNN_roi_pool(base_feat_t, rois_domain_t.view(-1, 5))

        # feed pooled features to top model
        # 将区域特征传入网络最后一个阶段，提取相应的特征(如resnet中的layer4)
        # 特征提取网络共划分为四个阶段，RPN前执行三个阶段，RPN后执行最后一个阶段
        # 对三组区域特征依次执行相同的操作(Chief Net、Ancillary Net以及用于计算实例级特征对齐损失的proposals)
        pooled_feat_s = self._head_to_tail(pooled_feat)
        pooled_feat_t = self._head_to_tail(pooled_feat_t)

        pooled_feat_domain_t = self._head_to_tail(pooled_feat_domain_t)

        # 预测边界框信息
        bbox_pred = self.RCNN_bbox_pred(pooled_feat_s)
        bbox_pred_t = self.RCNN_bbox_pred(pooled_feat_t)
        if self.training and not self.class_agnostic:
            # select the corresponding columns according to roi labels
            # ROI检测头为每个锚点均预测一组回归参数，一组回归参数包括所有类别的回归边界框
            # 即每个锚点对应num_classes * 4个数据
            # 这里根据label标签信息，选出特定类别的回归参数参与计算回归损失
            bbox_pred_view = bbox_pred.view(bbox_pred.size(0), int(bbox_pred.size(1) / 4), 4)
            bbox_pred_select = torch.gather(bbox_pred_view, 1,
                                            rois_label.view(rois_label.size(0), 1, 1).expand(rois_label.size(0), 1, 4))
            bbox_pred = bbox_pred_select.squeeze(1)

            bbox_pred_view_t = bbox_pred_t.view(bbox_pred_t.size(0), int(bbox_pred_t.size(1) / 4), 4)
            bbox_pred_select_t = torch.gather(bbox_pred_view_t, 1,
                                              rois_label_t.view(rois_label_t.size(0), 1, 1).expand(rois_label_t.size(0),
                                                                                                   1, 4))
            bbox_pred_t = bbox_pred_select_t.squeeze(1)

        # compute object classification probability
        # 预测类别概率
        cls_score = self.RCNN_cls_score(pooled_feat_s)
        cls_score_t = self.RCNN_cls_score(pooled_feat_t)
        # 测试的时候只利用Chief Net网络的分支进行测试，因此只需要对一个预测执行softmax
        cls_prob = F.softmax(cls_score, 1)

        RCNN_loss_cls = 0
        RCNN_loss_bbox = 0

        if self.training:
            # 依次计算分类损失和边界框回归损失
            # classification loss
            RCNN_loss_cls_s = F.cross_entropy(cls_score, rois_label)
            RCNN_loss_cls_t = F.cross_entropy(cls_score_t, rois_label_t)
            RCNN_loss_cls = RCNN_loss_cls_s + RCNN_loss_cls_t

            # bounding box regression L1 loss
            RCNN_loss_bbox_s = _smooth_l1_loss(bbox_pred, rois_target, rois_inside_ws, rois_outside_ws)
            RCNN_loss_bbox_t = _smooth_l1_loss(bbox_pred_t, rois_target_t, rois_inside_ws_t, rois_outside_ws_t)
            RCNN_loss_bbox = RCNN_loss_bbox_s + RCNN_loss_bbox_t

        # 测试阶段只利用Chief Net网络分支得到的预测数据
        cls_prob = cls_prob.view(batch_size, rois_t.size(1), -1)
        bbox_pred = bbox_pred.view(batch_size, rois_t.size(1), -1)

        # 后面执行对目标域数据的操作
        """ =================== for target =========================="""
        # 得到输入数据，变量所代表的含义与源域数据类似
        tgt_batch_size = tgt_im_data.size(0)
        tgt_im_info = tgt_im_info.data  # (size1,size2, image ratio(new image / source image) )
        tgt_gt_boxes = tgt_gt_boxes.data
        tgt_num_boxes = tgt_num_boxes.data
        tgt_need_backprop = tgt_need_backprop.data

        # feed image data to base model to obtain base feature map
        # 将目标域数据传入Chief Net网络，依次提取特征
        tgt_conv3_feat = self.conv3_s(tgt_im_data)
        tgt_conv4_feat = self.conv34_s(tgt_conv3_feat)
        tgt_base_feat = self.conv45_s(tgt_conv4_feat)

        # feed base feature map tp RPN to obtain rois
        # 将特征图传入RPN模块，得到proposals
        self.RCNN_rpn.eval()
        cfg.TEST.RPN_POST_NMS_TOP_N = rois_domain.size(1)
        tgt_rois, tgt_rpn_loss_cls, tgt_rpn_loss_bbox, tgt_anchor_score = \
            self.RCNN_rpn(tgt_base_feat, tgt_im_info, tgt_gt_boxes, tgt_num_boxes)

        tgt_rois_label = torch.ones(tgt_rois.size(1))

        # if it is training phrase, then use ground trubut bboxes for refining

        tgt_rois_label = None
        tgt_rois_target = None
        tgt_rois_inside_ws = None
        tgt_rois_outside_ws = None
        tgt_rpn_loss_cls = 0
        tgt_rpn_loss_bbox = 0

        tgt_rois = Variable(tgt_rois)
        # do roi pooling based on predicted rois
        # 利用proposals对原特征图执行roi pooling操作
        if cfg.POOLING_MODE == 'crop':
            tgt_grid_xy = _affine_grid_gen(tgt_rois.view(-1, 5), tgt_base_feat.size()[2:], self.grid_size)
            tgt_grid_yx = torch.stack([tgt_grid_xy.data[:, :, :, 1], tgt_grid_xy.data[:, :, :, 0]], 3).contiguous()
            tgt_pooled_feat = self.RCNN_roi_crop(tgt_base_feat, Variable(tgt_grid_yx).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                tgt_pooled_feat = F.max_pool2d(tgt_pooled_feat, 2, 2)
        elif cfg.POOLING_MODE == 'align':
            tgt_pooled_feat = self.RCNN_roi_align(tgt_base_feat, tgt_rois.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            tgt_pooled_feat = self.RCNN_roi_pool(tgt_base_feat, tgt_rois.view(-1, 5))

        # feed pooled features to top model
        # 将pooling得到的特征传入网络最后的阶段，与源域作用类似
        tgt_pooled_feat = self._head_to_tail(tgt_pooled_feat)
        # 预测分类分数，但后面没有用到
        tgt_cls_score = self.RCNN_cls_score(tgt_pooled_feat)
        # 后续没用到这个变量，因为训练阶段的目标域数据无标签
        tgt_cls_prob = F.softmax(tgt_cls_score, 1)

        # 计算域适应损失，首先计算源域数据的域适应损失
        # 注意，这里域适应损失只利用辅助网中的源域特征和主网中的目标域特征计算
        # 主网中的源域特征不参与域适应损失的计算
        """  DA loss   """
        # 初始化损失变量
        # DA LOSS
        # 源域域适应损失
        DA_img_loss_cls = 0
        DA_ins_loss_cls = 0
        # 目标域域适应损失
        tgt_DA_img_loss_cls = 0
        tgt_DA_ins_loss_cls = 0

        # 三个阶段的图像级领域对齐
        # DA conv3 针对conv3的域适应模块
        # 依次返回领域预测分数和对应的标签
        conv3_score, conv3_label = self.RCNN_imageDA_3(conv3_feat_t, need_backprop)
        # 预测分数经过一层softmax。其实这里不经过softmax也行，直接传入交叉熵损失
        conv3_prob = F.log_softmax(conv3_score, dim=1)
        # 计算损失，预测分数经过了softmax，再利用nll_loss计算损失
        DA_img_loss_conv3 = F.nll_loss(conv3_prob, conv3_label, ignore_index=-1)

        # conv4和conv5两个阶段的图像级域适应模块流程与conv3类似
        # DA conv4
        conv4_score, conv4_label = self.RCNN_imageDA_4(conv4_feat_t, need_backprop)

        conv4_prob = F.log_softmax(conv4_score, dim=1)
        DA_img_loss_conv4 = F.nll_loss(conv4_prob, conv4_label, ignore_index=-1)

        # DA conv5
        base_score, base_label = self.RCNN_imageDA(base_feat_t, need_backprop)

        base_prob = F.log_softmax(base_score, dim=1)
        DA_img_loss_conv5 = F.nll_loss(base_prob, base_label, ignore_index=-1)
        # 求和得到总的图像级域适应损失
        DA_img_loss_cls = DA_img_loss_conv3 + DA_img_loss_conv4 + DA_img_loss_conv5

        # instance DA 计算实例级域适应损失
        # 得到实例级领域预测分数和对应的标签
        instance_sigmoid, same_size_label = self.RCNN_instanceDA(pooled_feat_domain_t, need_backprop)
        # 初始化损失函数
        instance_loss = nn.BCELoss()
        # 计算损失
        DA_ins_loss_cls = instance_loss(instance_sigmoid, same_size_label)

        # 计算目标域数据的域适应损失，流程与源域类似
        """  ************** target loss ****************  """

        # DA conv3 target
        tgt_conv3_score, tgt_conv3_label = \
            self.RCNN_imageDA_3(tgt_conv3_feat, tgt_need_backprop)

        tgt_conv3_prob = F.log_softmax(tgt_conv3_score, dim=1)
        tgt_DA_img_loss_conv3 = F.nll_loss(tgt_conv3_prob, tgt_conv3_label, ignore_index=-1)

        # DA conv4 target
        tgt_conv4_score, tgt_conv4_label = \
            self.RCNN_imageDA_4(tgt_conv4_feat, tgt_need_backprop)

        tgt_conv4_prob = F.log_softmax(tgt_conv4_score, dim=1)
        tgt_DA_img_loss_conv4 = F.nll_loss(tgt_conv4_prob, tgt_conv4_label, ignore_index=-1)

        # DA conv5 target
        tgt_base_score, tgt_base_label = \
            self.RCNN_imageDA(tgt_base_feat, tgt_need_backprop)

        tgt_base_prob = F.log_softmax(tgt_base_score, dim=1)
        tgt_DA_img_loss_conv5 = F.nll_loss(tgt_base_prob, tgt_base_label, ignore_index=-1)
        tgt_DA_img_loss_cls = tgt_DA_img_loss_conv3 + tgt_DA_img_loss_conv4 + tgt_DA_img_loss_conv5

        # instance_target
        tgt_instance_sigmoid, tgt_same_size_label = \
            self.RCNN_instanceDA(tgt_pooled_feat, tgt_need_backprop)
        tgt_instance_loss = nn.BCELoss()

        tgt_DA_ins_loss_cls = tgt_instance_loss(tgt_instance_sigmoid, tgt_same_size_label)
        # 返回数据
        return rois_t, cls_prob, bbox_pred, rpn_loss_cls, rpn_loss_bbox, RCNN_loss_cls, RCNN_loss_bbox, rois_label_t, \
               DA_img_loss_cls, tgt_DA_img_loss_cls, DA_ins_loss_cls, tgt_DA_ins_loss_cls

    图像级域分类器¶

    class _ImageDA(nn.Module):
    def __init__(self, dim):
        super(_ImageDA, self).__init__()
        self.dim = dim
        # 定义好卷积层以及激活层
        self.Conv1 = nn.Conv2d(self.dim, 512, kernel_size=1, stride=1, bias=False)
        self.Conv2 = nn.Conv2d(512, 2, kernel_size=1, stride=1, bias=False)
        self.reLu = nn.ReLU(inplace=False)
        # 定义标签生成器
        self.LabelResizeLayer = ImageLabelResizeLayer()

    def forward(self, x, need_backprop):
        # 套一个梯度反转模块
        x = grad_reverse(x, need_backprop)
        # conv1->relu->conv2
        x = self.reLu(self.Conv1(x))
        x = self.Conv2(x)
        # 生成对应的标签
        label = self.LabelResizeLayer(x, need_backprop)
        return x, label

    实例级域分类器¶

    class _InstanceDA(nn.Module):
    def __init__(self):
        super(_InstanceDA, self).__init__()
        # 定义好线性回归层、激活层、dropout层
        self.dc_ip1 = nn.Linear(4096, 1024)
        self.dc_relu1 = nn.ReLU()
        self.dc_drop1 = nn.Dropout(p=0.5)

        self.dc_ip2 = nn.Linear(1024, 1024)
        self.dc_relu2 = nn.ReLU()
        self.dc_drop2 = nn.Dropout(p=0.5)

        self.clssifer = nn.Linear(1024, 1)
        # 定义标签生成器
        self.LabelResizeLayer = InstanceLabelResizeLayer()

    def forward(self, x, need_backprop):
        # 套一个梯度反转模块
        x = grad_reverse(x, need_backprop)
        # fc->relu->dropout
        x = self.dc_drop1(self.dc_relu1(self.dc_ip1(x)))
        x = self.dc_drop2(self.dc_relu2(self.dc_ip2(x)))
        # sigmoid归一化操作
        x = F.sigmoid(self.clssifer(x))
        # 生成对应的标签
        label = self.LabelResizeLayer(x, need_backprop)
        return x, label
