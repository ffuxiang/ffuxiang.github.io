    域适应：DA Faster R-CNN¶

    综述¶

    会议时间：IEEE Conference on Computer Vision and Pattern Recognition 2018 (CVPR, 2018)

    论文地址：https://openaccess.thecvf.com/content_cvpr_2018/papers/Chen_Domain_Adaptive_Faster_CVPR_2018_paper.pdf

    源码地址：https://github.com/yuhuayc/da-faster-rcnn（caffe）、https://github.com/krumo/Domain-Adaptive-Faster-RCNN-PyTorch（PyTorch）

    针对领域：域适应目标检测

    网络结构¶

    整体结构与Faster R-CNN差不多，主要是添加了域适应模块以及DA损失的计算

    class GeneralizedRCNN(nn.Module):
    """
    Main class for Generalized R-CNN. Currently supports boxes and masks.
    It consists of three main parts:
    - backbone
    - rpn
    - heads: takes the features + the proposals from the RPN and computes
        detections / masks from it.
    """

    def __init__(self, cfg):
        super(GeneralizedRCNN, self).__init__()

        self.backbone = build_backbone(cfg)
        self.rpn = build_rpn(cfg)
        self.roi_heads = build_roi_heads(cfg)
        # 域适应模块
        self.da_heads = build_da_heads(cfg)

    def forward(self, images, targets=None):
        """
        Arguments:
            images (list[Tensor] or ImageList): images to be processed
            targets (list[BoxList]): ground-truth boxes present in the image (optional)

        Returns:
            result (list[BoxList] or dict[Tensor]): the output from the model.
                During training, it returns a dict[Tensor] which contains the losses.
                During testing, it returns list[BoxList] contains additional fields
                like `scores`, `labels` and `mask` (for Mask R-CNN models).

        """
        if self.training and targets is None:
            raise ValueError("In training mode, targets should be passed")
        images = to_image_list(images)
        features = self.backbone(images.tensors)
        proposals, proposal_losses = self.rpn(images, features, targets)
        da_losses = {}
        if self.roi_heads:
            x, result, detector_losses, da_ins_feas, da_ins_labels = self.roi_heads(features, proposals, targets)
            if self.da_heads:
                # 计算DA域分类对抗损失
                da_losses = self.da_heads(features, da_ins_feas, da_ins_labels, targets)
        else:
            # RPN-only models don't have roi_heads
            x = features
            result = proposals
            detector_losses = {}

        if self.training:
            losses = {}
            losses.update(detector_losses)
            losses.update(proposal_losses)
            losses.update(da_losses)
            return losses

        return result

    域适应模块¶

    class DomainAdaptationModule(torch.nn.Module):
    """
    Module for Domain Adaptation Component. Takes feature maps from the backbone and instance
    feature vectors, domain labels and proposals. Works for both FPN and non-FPN.
    """

    def __init__(self, cfg):
        super(DomainAdaptationModule, self).__init__()

        self.cfg = cfg.clone()

        stage_index = 4
        stage2_relative_factor = 2 ** (stage_index - 1)
        res2_out_channels = cfg.MODEL.RESNETS.RES2_OUT_CHANNELS
        num_ins_inputs = cfg.MODEL.ROI_BOX_HEAD.MLP_HEAD_DIM if cfg.MODEL.BACKBONE.CONV_BODY.startswith('V') else res2_out_channels * stage2_relative_factor

        self.resnet_backbone = cfg.MODEL.BACKBONE.CONV_BODY.startswith('R')
        # 平均池化，用于压缩特征
        self.avgpool = nn.AvgPool2d(kernel_size=7, stride=7)
        # 各个损失的权重
        self.img_weight = cfg.MODEL.DA_HEADS.DA_IMG_LOSS_WEIGHT
        self.ins_weight = cfg.MODEL.DA_HEADS.DA_INS_LOSS_WEIGHT
        self.cst_weight = cfg.MODEL.DA_HEADS.DA_CST_LOSS_WEIGHT
        # 这里权重传入-0.1，负数表示梯度反转，0.1表示最终的损失函数变为0.1倍，与原文λ的设置有关，具体请看GradientScalarLayer类
        self.grl_img = GradientScalarLayer(-1.0*self.cfg.MODEL.DA_HEADS.DA_IMG_GRL_WEIGHT)  # 这里乘以-1，表示用于梯度取反
        self.grl_ins = GradientScalarLayer(-1.0*self.cfg.MODEL.DA_HEADS.DA_INS_GRL_WEIGHT)
        # 这里默认乘以0.1，同上表示回传梯度的时候梯度变为0.1，与上面不同的是这里是正数，梯度不做反转
        # 这里主要用于计算正则化损失
        self.grl_img_consist = GradientScalarLayer(1.0*self.cfg.MODEL.DA_HEADS.DA_IMG_GRL_WEIGHT)
        self.grl_ins_consist = GradientScalarLayer(1.0*self.cfg.MODEL.DA_HEADS.DA_INS_GRL_WEIGHT)
        # 特征图通道数
        in_channels = cfg.MODEL.BACKBONE.OUT_CHANNELS
        # 定义图片级域分类器
        self.imghead = DAImgHead(in_channels)
        # 定义实例级域分类器
        self.inshead = DAInsHead(num_ins_inputs)
        # 定义DA损失函数
        self.loss_evaluator = make_da_heads_loss_evaluator(cfg)
    # img_features整张图片特征、da_ins_feature实例级图片特征、da_ins_labels实例级域标签

    def forward(self, img_features, da_ins_feature, da_ins_labels, targets=None):
        """
        Arguments:
            img_features (list[Tensor]): features computed from the images that are
                used for computing the predictions.
            da_ins_feature (Tensor): instance-level feature vectors
            da_ins_labels (Tensor): domain labels for instance-level feature vectors
            targets (list[BoxList): ground-truth boxes present in the image (optional)

        Returns:
            losses (dict[Tensor]): the losses for the model during training. During
                testing, it is an empty dict.
        """
        if self.resnet_backbone:
            da_ins_feature = self.avgpool(da_ins_feature)  # 这里把预测到的目标的特征拉伸
        # 将得到的特征拉直，尺寸为(batch*ROI检测头，通道数量)
        da_ins_feature = da_ins_feature.view(da_ins_feature.size(0), -1)  # 拉直
        # 这里将特征图套入grl_img，表示梯度反转，并且乘以0.1，将图像级特征和实例级特征均做一个反转
        img_grl_fea = [self.grl_img(fea) for fea in img_features]
        ins_grl_fea = self.grl_ins(da_ins_feature)
        # 这里将特征套入grl_img_consist，梯度变为0.1。同上，将两个级别的特征均套进去
        img_grl_consist_fea = [self.grl_img_consist(fea) for fea in img_features]
        ins_grl_consist_fea = self.grl_ins_consist(da_ins_feature)
        # 经过图片级域分类器，得到预测结果
        da_img_features = self.imghead(img_grl_fea)
        # 经过实例级域分类器，得到预测结果
        da_ins_features = self.inshead(ins_grl_fea)
        # da_img_consist_features和da_ins_consist_features用于计算正则化损失，梯度无需反转
        da_img_consist_features = self.imghead(img_grl_consist_fea)
        da_ins_consist_features = self.inshead(ins_grl_consist_fea)
        # 求正则化损失时，预测概率需要先经过一个sigmoid
        da_img_consist_features = [fea.sigmoid() for fea in da_img_consist_features]
        da_ins_consist_features = da_ins_consist_features.sigmoid()
        if self.training:
            # 依次将梯度反转后的域分类器分类结果、梯度反转前的域分类器分类结果、实例级标签、总标签传入损失函数，得到DA损失
            da_img_loss, da_ins_loss, da_consistency_loss = self.loss_evaluator(
                da_img_features, da_ins_features, da_img_consist_features, da_ins_consist_features, da_ins_labels, targets
            )
            losses = {}
            if self.img_weight > 0:
                losses["loss_da_image"] = self.img_weight * da_img_loss
            if self.ins_weight > 0:
                losses["loss_da_instance"] = self.ins_weight * da_ins_loss
            if self.cst_weight > 0:
                losses["loss_da_consistency"] = self.cst_weight * da_consistency_loss
            return losses
        return {}

    图像级域分类器¶

    一组特征图生成一张图

    class DAImgHead(nn.Module):
    """
    Adds a simple Image-level Domain Classifier head
    图像级域分类器
    """

    def __init__(self, in_channels):
        """
        Arguments:
            in_channels (int): number of channels of the input feature
        """
        super(DAImgHead, self).__init__()
        # 特征图经过两层卷积，变为一张图，通道数为1
        # 该图上的数据表示特征图每个位置的领域分类情况
        self.conv1_da = nn.Conv2d(in_channels, 512, kernel_size=1, stride=1)
        self.conv2_da = nn.Conv2d(512, 1, kernel_size=1, stride=1)
        # 卷积层参数初始化
        for l in [self.conv1_da, self.conv2_da]:
            torch.nn.init.normal_(l.weight, std=0.001)
            torch.nn.init.constant_(l.bias, 0)
    # 前向传播过程
    def forward(self, x):
        img_features = []
        # 依次经过:卷积层1->relu->卷积层2，生成一张图
        # 按特征层数遍历(FPN利用了多层特征进行目标检测)
        for feature in x:
            t = F.relu(self.conv1_da(feature))
            img_features.append(self.conv2_da(t))
        # 返回域分类图，每层均得到一个图
        return img_features

    实例级域分类器¶

    每个实例生成一个数

    class DAInsHead(nn.Module):
    """
    Adds a simple Instance-level Domain Classifier head
    实例级域分类器
    """

    def __init__(self, in_channels):
        """
        Arguments:
            in_channels (int): number of channels of the input feature
        """
        super(DAInsHead, self).__init__()
        # 定义三个线性回归层，最后输出一个值，表示该实例的领域分类情况
        self.fc1_da = nn.Linear(in_channels, 1024)
        self.fc2_da = nn.Linear(1024, 1024)
        self.fc3_da = nn.Linear(1024, 1)
        # 初始化线性回归层参数
        for l in [self.fc1_da, self.fc2_da]:
            nn.init.normal_(l.weight, std=0.01)
            nn.init.constant_(l.bias, 0)
        nn.init.normal_(self.fc3_da.weight, std=0.05)
        nn.init.constant_(self.fc3_da.bias, 0)

    def forward(self, x):
        # 输入的特征x已经被拉伸成一个向量
        # 特征x依次经过fc、relu、dropout
        # 每个实例均生成一个数
        x = F.relu(self.fc1_da(x))
        x = F.dropout(x, p=0.5, training=self.training)

        x = F.relu(self.fc2_da(x))
        x = F.dropout(x, p=0.5, training=self.training)

        x = self.fc3_da(x)
        # 返回域分类概率
        return x

    GRL模块¶

    梯度反转模块，领域分类损失产生的梯度传到特征提取网格时需要做一个反转，即变为负数

    class GradientScalarLayer(torch.nn.Module):
    def __init__(self, weight):
        super(GradientScalarLayer, self).__init__()
        self.weight = weight

    def forward(self, input):
        return gradient_scalar(input, self.weight)

gradient_scalar = _GradientScalarLayer.apply

class _GradientScalarLayer(torch.autograd.Function):
    @staticmethod
    def forward(ctx, input, weight):
        ctx.weight = weight
        return input.view_as(input)

    @staticmethod
    def backward(ctx, grad_output):
        grad_input = grad_output.clone()
        # 反向传播过程中，梯度传到这一层时，会与参数weight相乘
        # 如果此时weight为负数，则梯度变为原来的负数
        return ctx.weight*grad_input, None

    DA损失¶

    class DALossComputation(object):
    """
    This class computes the DA loss.
    """

    def __init__(self, cfg):
        self.cfg = cfg.clone()
        resolution = cfg.MODEL.ROI_BOX_HEAD.POOLER_RESOLUTION
        scales = cfg.MODEL.ROI_BOX_HEAD.POOLER_SCALES
        sampling_ratio = cfg.MODEL.ROI_BOX_HEAD.POOLER_SAMPLING_RATIO
        pooler = Pooler(
            output_size=(resolution, resolution),
            scales=scales,
            sampling_ratio=sampling_ratio,
        )

        self.pooler = pooler
        self.avgpool = nn.AvgPool2d(kernel_size=resolution, stride=resolution)

    def prepare_masks(self, targets):
        masks = []
        for targets_per_image in targets:
            is_source = targets_per_image.get_field('is_source')
            mask_per_image = is_source.new_ones(1, dtype=torch.uint8) if is_source.any() else is_source.new_zeros(1, dtype=torch.uint8)
            masks.append(mask_per_image)
        return masks

    def __call__(self, da_img, da_ins, da_img_consist, da_ins_consist, da_ins_labels, targets):
        """
        Arguments:
            da_img (list[Tensor]) 用于计算图片级域分类器损失
            da_img_consist (list[Tensor]) 用于计算正则化损失
            da_ins (Tensor) 用于计算实例级域分类器损失
            da_ins_consist (Tensor) 用于计算正则化损失
            da_ins_labels (Tensor) 用于计算实例级域分类器损失
            targets (list[BoxList])

        Returns:
            da_img_loss (Tensor)
            da_ins_loss (Tensor)
            da_consist_loss (Tensor)
        """

        masks = self.prepare_masks(targets)
        masks = torch.cat(masks, dim=0)

        da_img_flattened = []
        da_img_labels_flattened = []
        # for each feature level, permute the outputs to make them be in the
        # same format as the labels. Note that the labels are computed for
        # all feature levels concatenated, so we keep the same representation
        # for the image-level domain alignment
        # 按特征层数遍历
        for da_img_per_level in da_img:
            # 得到特征图尺寸
            N, A, H, W = da_img_per_level.shape
            # 转换维度，通道维度移到最后
            da_img_per_level = da_img_per_level.permute(0, 2, 3, 1)
            # 初始化标签
            da_img_label_per_level = torch.zeros_like(da_img_per_level, dtype=torch.float32)
            # 如果图片来自目标域，则标签设置为1
            da_img_label_per_level[masks, :] = 1
            # 图像级预测拉直，从一张图拉值为一个向量
            da_img_per_level = da_img_per_level.reshape(N, -1)
            da_img_label_per_level = da_img_label_per_level.reshape(N, -1)
            # 储存预测和标签，用于后续的损失计算
            da_img_flattened.append(da_img_per_level)
            da_img_labels_flattened.append(da_img_label_per_level)
        # 合并数据
        da_img_flattened = torch.cat(da_img_flattened, dim=0)
        da_img_labels_flattened = torch.cat(da_img_labels_flattened, dim=0)
        # 计算图像级域分类器损失
        da_img_loss = F.binary_cross_entropy_with_logits(
            da_img_flattened, da_img_labels_flattened
        )
        # 计算实例级域分类器损失
        da_ins_loss = F.binary_cross_entropy_with_logits(
            torch.squeeze(da_ins), da_ins_labels.type(torch.cuda.FloatTensor)
        )
        # 计算正则化损失
        da_consist_loss = consistency_loss(da_img_consist, da_ins_consist, da_ins_labels, size_average=True)

        return da_img_loss, da_ins_loss, da_consist_loss

    正则化损失¶

    def consistency_loss(img_feas, ins_fea, ins_labels, size_average=True):
    """
    Consistency regularization as stated in the paper
    `Domain Adaptive Faster R-CNN for Object Detection in the Wild`
    L_cst = \sum_{i,j}||\frac{1}{|I|}\sum_{u,v}p_i^{(u,v)}-p_{i,j}||_2
    """
    loss = []
    # 实例级预测数量，检测头数量乘以batch
    len_ins = ins_fea.size(0)
    # 第一个表示实例来自目标域的数量(ins_labels非零)、第二个表示来源域的数量
    intervals = [torch.nonzero(ins_labels).size(0), len_ins - torch.nonzero(ins_labels).size(0)]
    # 按特征层数遍历
    for img_fea_per_level in img_feas:
        # 得到图像级预测(一张图)的尺寸，N表示batch
        N, A, H, W = img_fea_per_level.shape
        # 求得图像级预测的平均值，得到两个均值
        img_fea_per_level = torch.mean(img_fea_per_level.reshape(N, -1), 1)
        img_feas_per_level = []
        # 这里限制了训练batch为2，一张源域图像，一张目标域图像
        # 具体请看论文中实施细节
        assert N==2, \
            "only batch size=2 is supported for consistency loss now, received batch size: {}".format(N)
        # 按batch遍历
        for i in range(N):
            # 将均值复制，有几个对应领域的实例级就将均值复制几次(按领域数量复制的)
            img_fea_mean = img_fea_per_level[i].view(1, 1).repeat(intervals[i], 1)
            img_feas_per_level.append(img_fea_mean)
        # 合并之前复制的均值
        img_feas_per_level = torch.cat(img_feas_per_level, dim=0)
        # 均值与领域预测值做差，取绝对值。但论文用的是l2距离(欧氏距离)、源码用的l1距离(绝对值距离)
        loss_per_level = torch.abs(img_feas_per_level - ins_fea)
        loss.append(loss_per_level)
    # 合并再取均值，得到最后的损失
    loss = torch.cat(loss, dim=1)
    if size_average:
        return loss.mean()
    return loss.sum()
