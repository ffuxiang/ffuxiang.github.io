    前向传播¶

    注：由于目标域缺少标签，因此在目标域图片的训练过程中无法筛选出特定比例的正负样本参与训练，源码中直接按分数排序，选取了前256个。

    # domain_label源域表示1，目标域表示0
    def forward(self, im_data, im_info, gt_boxes, num_boxes, domain_label, imdb=None):

        if not global_variable.ProcessDontCare and global_variable.ImdbIsKitti:
            # ignore dontcare when imdb is kitti
            # pdb.set_trace()
            gt_boxes[gt_boxes[:, :, -1] == 3] = 0
            gt_boxes = torch.cat((gt_boxes[gt_boxes[:, :, -1] != 0], gt_boxes[gt_boxes[:, :, -1] == 0]), 0).view(1, -1,
                                                                                                                 5)

        global counter
        # 得到batch size尺寸
        batch_size = im_data.size(0)
        #
        im_info = im_info.data
        gt_boxes = gt_boxes.data
        num_boxes = num_boxes.data

        # 将图像传入基线网络进行提取特征，这里分为了两个阶段
        # 以resnet为例，第一阶段(base1)是开始到layer3，第二阶段(base2)是layer4
        # 第一阶段的输出主要用于局部特征对齐，即文中的3.4小节
        local_base_feat = self.RCNN_base1(im_data)
        base_feat = self.RCNN_base2(local_base_feat)

        # 将特征传入RPN模块，得到感兴趣的区域，根据图片领域的不同，会返回不同的结果
        if domain_label == 1:  # source
            # 依次返回区域建议proposal、rpn分类(前后景)、rpn回归损失、proposal对应的预测分数(已切断梯度)、带有梯度的前景预测分数(主要用于计算局部对齐模块的权重)
            rois_before, rpn_loss_cls, rpn_loss_bbox, roi_scores_before, rpn_cls_prob = self.RCNN_rpn(base_feat,
                                                                                                      im_info, gt_boxes,
                                                                                                      num_boxes,
                                                                                                      domain_label)
        elif domain_label == 0:  # target
            # rpn_cls_score表示rpn中参与损失计算的前后景预测分数(按特定正负样本比例筛选后的分数)
            # rpn_cls_score_reshape表示reshape后的预测分数，尺寸为[锚点数量，2],2代表前后景分数
            # rpn_bbox_pred表示rpn预测的边界框回归参数
            # roi_scores_with_grad表示带有梯度的proposal分数
            rois_before, rpn_loss_cls, rpn_loss_bbox, roi_scores_before, \
            rpn_cls_score, rpn_cls_score_reshape, batch_size, rpn_bbox_pred, rpn_cls_prob, roi_scores_with_grad \
                = self.RCNN_rpn(base_feat, im_info, gt_boxes, num_boxes, domain_label)

        # if it is training phase, then use ground truth bboxes for refining
        if self.training:
            # 在局部对齐之前，先将特征传入梯度反转层
            reversed_local_feat = ReverseLayerF.apply(local_base_feat, 0.5)
            # 将浅层特征传入局部域分类器(像素级别)，得到领域预测图
            local_dis_output, _ = self.local_discriminator(reversed_local_feat)
            # 使用heat map global loss
            # 将域分类器结果、领域标签、前景预测图(对应论文中fi)传入local_loss_layer，计算域分类损失，与论文公式11、12对应
            domain_cls_loss = self.local_loss_layer(local_dis_output, domain_label, rpn_cls_prob)
            # local_dis_output_0网络中后面没有用到，直接在最后返回了
            local_dis_output_0 = local_dis_output.mean()

            # rpn filter and rpn target for cls and reg calculation
            if domain_label == 1:  # source
                # 源域数据，为每个proposal分配类别标签和回归参数，并且对锚点进行筛选(按特定的正负样本比例筛选)，筛选出参与RPC损失计算的proposal
                roi_data = self.RCNN_proposal_target(rois_before, gt_boxes, num_boxes)
                # rois表示参与ROI Pooling的区域、rois_label表示区域类别、rois_target表示区域对应的回归参数
                rois, rois_label, rois_target, rois_inside_ws, rois_outside_ws = roi_data
            else:  # target
                # 目标域数据，由于目标域数据缺少标签，因此无法对预测的ROI进行筛选，因此目标域直接选取前256个锚点对应的roi视为proposal
                # 按预测分数排序，对预测分数先执行了nms，执行后得到的数据为排序后的数据，再选取前256个参与RPC的运算，对应下面的roi
                roi_data = self.RCNN_proposal_target_for_target(rois_before, gt_boxes, num_boxes, roi_scores_before,
                                                                roi_scores_with_grad)
                # roi_scores表示roi对应的预测分数，用于计算目标域数据的RPC损失(起到加权的作用)，这里已经切断梯度了
                # roi_scores_grad表示带有梯度的roi预测分数，用于计算MDC损失
                # 目标域样本无法利用rois_label、rois_target，因为没有标签
                rois, rois_label, rois_target, rois_inside_ws, rois_outside_ws, roi_scores, roi_scores_grad = roi_data

            rois_label = Variable(rois_label.view(-1).long())
            rois_target = Variable(rois_target.view(-1, rois_target.size(2)))
            rois_inside_ws = Variable(rois_inside_ws.view(-1, rois_inside_ws.size(2)))
            rois_outside_ws = Variable(rois_outside_ws.view(-1, rois_outside_ws.size(2)))
        else:
            rois_label = None
            rois_target = None
            rois_inside_ws = None
            rois_outside_ws = None
            rpn_loss_cls = 0
            rpn_loss_bbox = 0
            domain_cls_loss = 0
            local_dis_output_0 = 0
        # 如果是训练阶段，则选择之前筛选好的roi；如果是测试阶段，则没法进行筛选，直接利用筛选前的roi
        rois = Variable(rois) if self.training else Variable(rois_before)
        # do roi pooling based on predicted rois
        # 不同策略的ROI Pooling
        if cfg.POOLING_MODE == 'crop':
            # pdb.set_trace()
            # pooled_feat_anchor = _crop_pool_layer(base_feat, rois.view(-1, 5))
            grid_xy = _affine_grid_gen(rois.view(-1, 5), base_feat.size()[2:], self.grid_size)
            grid_yx = torch.stack([grid_xy.data[:, :, :, 1], grid_xy.data[:, :, :, 0]], 3).contiguous()
            pooled_feat = self.RCNN_roi_crop(base_feat, Variable(grid_yx).detach())
            if cfg.CROP_RESIZE_WITH_MAX_POOL:
                pooled_feat = F.max_pool2d(pooled_feat, 2, 2)
        elif cfg.POOLING_MODE == 'align':
            pooled_feat = self.RCNN_roi_align(base_feat, rois.view(-1, 5))
        elif cfg.POOLING_MODE == 'pool':
            pooled_feat = self.RCNN_roi_pool(base_feat, rois.view(-1, 5))

        # feed pooled features to top model，得到每个区域的特征
        pooled_feat = self._head_to_tail(pooled_feat)

        # compute bbox offset
        # 预测RPC中边界框回归参数
        bbox_pred = self.RCNN_bbox_pred(pooled_feat)
        if self.training and not self.class_agnostic:
            # select the corresponding columns according to roi labels
            bbox_pred_view = bbox_pred.view(bbox_pred.size(0), int(bbox_pred.size(1) / 4), 4)
            bbox_pred_select = torch.gather(bbox_pred_view, 1,
                                            rois_label.view(rois_label.size(0), 1, 1).expand(rois_label.size(0), 1, 4))
            bbox_pred = bbox_pred_select.squeeze(1)

        # compute object classification probability
        # 预测类别分数
        cls_score = self.RCNN_cls_score(pooled_feat)
        cls_prob = F.softmax(cls_score, 1)

        rpn_cls_align_loss = 0

        # 目标域数据的训练阶段
        if domain_label == 0 and self.training:
            # cls entropy minimization CLS，用于训练RPC模块的最小熵值损失，对应论文公式(4)
            # 传入预测的类别分数、roi分数(即RPN预测的proposal分数)
            rpn_cls_align_loss = self.loss_rpn_cls_layer(cls_score, roi_scores.squeeze())

        RCNN_loss_cls = 0
        RCNN_loss_bbox = 0
        # 源域数据的训练阶段
        if self.training and domain_label == 1:
            # classification loss，分类损失
            if global_variable.ProcessDontCare:
                RCNN_loss_cls = F.cross_entropy(cls_score, rois_label, weight=torch.FloatTensor([1, 1, 1, 0]).cuda())
            else:
                RCNN_loss_cls = F.cross_entropy(cls_score, rois_label)

            # 边界框回归损失
            RCNN_loss_bbox = _smooth_l1_loss(bbox_pred, rois_target, rois_inside_ws, rois_outside_ws)

        cls_prob = cls_prob.view(batch_size, rois.size(1), -1)
        bbox_pred = bbox_pred.view(batch_size, rois.size(1), -1)

        # 目标域的训练阶段
        if domain_label == 0 and self.training:

            # weight version，传入proposal(筛选后的roi)以及对应的类别预测分数
            # 利用RPC的预测结果对RPN中的预测打标签
            gt_boxes, gt_num, gf_boxes, gf_num, gt_score, gf_score = generate_boxes_from_cls_score(rois, cls_prob, imdb)
            counter += 1
            if gt_num > 0 and gf_num > 0:
                gt_boxes = gt_boxes.cuda()
                gf_boxes = gf_boxes.cuda()
                # 计算目标域中RPN的分类损失和边界框回归损失
                rpn_loss_cls, rpn_loss_bbox = self.rpn_training_target(gt_boxes, im_info, num_boxes, domain_label,
                                                                       gf_boxes, rpn_cls_score, rpn_cls_score_reshape,
                                                                       batch_size, rpn_bbox_pred, gt_score, gf_score)
            else:
                print("gt num : {}, gf num : {}".format(gt_num, gf_num))
                rpn_loss_cls = 0
                rpn_loss_bbox = 0
        # 目标域的训练阶段
        if domain_label == 0 and self.training:
            # 计算MDC损失，对应论文公式10，输入RPC中的类别预测分数和RPN中的前后景预测分数
            # 依次得到两种损失，分别对应于特征提取器F(最小化MDC)，和RPN、RPC中的分类模块(最大化MDC)
            discrepancy_loss_min, discrepancy_loss_max = self.loss_for_discrepancy(cls_score, roi_scores_grad.squeeze())
        else:
            discrepancy_loss_min = 0
            discrepancy_loss_max = 0

        return rois, cls_prob, bbox_pred, rpn_loss_cls, rpn_loss_bbox, RCNN_loss_cls, RCNN_loss_bbox, rois_label, domain_cls_loss, local_dis_output_0, rpn_cls_align_loss, discrepancy_loss_min, discrepancy_loss_max

    # 像素级别的局部域分类器
class LocalDiscriminator(nn.Module):
    def __init__(self, in_channels):
        super(LocalDiscriminator, self).__init__()
        # 依次定义三层卷积，最后一层输出通道数为1
        self.conv1 = nn.Conv2d(in_channels=in_channels, out_channels=256, kernel_size=1, stride=1, padding=0)
        self.conv2 = nn.Conv2d(in_channels=256, out_channels=128, kernel_size=1, stride=1, padding=0)
        self.conv3 = nn.Conv2d(in_channels=128, out_channels=1, kernel_size=1, stride=1, padding=0)

    def forward(self, input):
        # 依次经过卷积，relu
        x = input
        x = self.conv1(x)
        x = F.relu(x)
        x = self.conv2(x)
        x = F.relu(x)
        # context_vector后面没用到
        context_vector = x.detach()
        x = self.conv3(x)
        # 最后经过sigmoid归一化操作，得到预测概率
        x = F.sigmoid(x)
        # 返回预测概率
        return x, context_vector

    # RPC模块协同训练损失
class LossForRPNCLS(nn.Module):
    def __init__(self, power):
        super(LossForRPNCLS, self).__init__()
        # 定义参数lambda，对应公式6中的指数，默认5
        self.power = power

    def forward(self, cls_score, roi_score):
        # 下述三行代码相当于计算公式(6)
        # 按roi_score与0.5的相对大小分开计算，相当于将公式(6)中的绝对值拆开
        factor_bigger_half = (2 * roi_score - 1).pow(self.power)
        factor_smaller_half = (1 - 2 * roi_score).pow(self.power)
        factor = (roi_score > 0.5).float() * factor_bigger_half + (roi_score <= 0.5).float() * factor_smaller_half
        # 得到权重fw，注意这里切断了梯度，即这个损失只用于更新RPC模块，不更新RPN模块
        factor = factor.detach()
        # 计算熵值，对应论文中公式(5)——概率乘以对数概率
        ent = F.softmax(cls_score, dim=1) * F.log_softmax(cls_score, dim=1)
        ent = -1.0 * ent.sum(dim=1)
        # 权重与熵值相乘，得到公式(4)
        return ent * factor

    MCD损失¶

    # 计算MDC损失
class LossForDiscrepancy(nn.Module):
    def __init__(self):
        super(LossForDiscrepancy, self).__init__()

    def forward(self, cls_score, roi_score):
        # 将分数经过一层softmax归一化操作
        cls_prob = F.softmax(cls_score, 1)
        # 计算权重，对应论文中公式9
        weight = torch.min(2 * torch.min(1 - cls_prob[:, 0], cls_prob[:, 0]),
                           2 * torch.min(roi_score[:], 1 - roi_score[:]))
        # 切断权重所带来的梯度，表明MDC损失产生的梯度不会从权重weight传入到RPN和RPC中
        weight = weight.detach()
        # 权重开平方处理，这里2对应论文里的lambda
        weight = weight.pow(2)
        # 计算损失，对应论文公式10
        loss_to_min = (1 - cls_prob[:, 0] - roi_score[:]).abs() * weight
        # 注意这个取负值的损失用于更新RPN和RPC中的前后景分类模块，因为分类模块需要最大化该损失，因此需要乘一个负值
        # 这里也可以利用梯度反转层实现
        loss_to_max = -(1 - cls_prob[:, 0] - roi_score[:]).abs() * weight
        # 依次返回两种损失，分别对应于特征提取器F(最小化MDC)，和RPN、RPC中的分类模块(最大化MDC)
        return loss_to_min, loss_to_max

    # 局部对齐模块的损失
class LossForLocal(nn.Module):
    def __init__(self):
        super(LossForLocal, self).__init__()

    def forward(self, input, domain, heat_map=None):
        y = input
        if domain == 1:
            # 源域数据
            loss = y.pow(2)
        elif domain == 0:
            # 目标域数据
            loss = (1 - y).pow(2)
        else:
            raise Exception("domain label must be consistent in one batch")
        if heat_map is None:
            return loss
        else:
            # 这里取出所有锚点的前景分数预测图，注意要切断梯度，这里的9表示特征图中每个像素点会对应9个锚点
            heat_map = heat_map[0][9:, :, :].detach()
            # 所有分数图沿第一维度求和，之后维度转化为二维
            heat_map = heat_map.sum(0).cpu().view(heat_map.shape[1], heat_map.shape[2]).numpy()
            # 将预测图尺寸调整为与损失相同的尺寸
            heat_map = resize(heat_map, (loss.shape[2], loss.shape[3]))
            # 预测图数据与损失乘积，得到最终的对抗损失
            return loss * torch.FloatTensor(heat_map).view(1, 1, heat_map.shape[0], heat_map.shape[1]).cuda()
