    class ACNet(nn.Module):

    def __init__(self, config):
        super(ACNet, self).__init__()
        # 加载ACNet网络的各种参数
        self.target_size = (3, config['target_size'], config['target_size'])
        self.class_num = config['class_num']
        self.aux_conv_in = config['aux_conv_in']
        self.tree_in = config['tree_in']
        self.use_cuda = config['use_cuda']
        self.log_dir = config['log_dir']
        self.tree_height = config['tree_height']

        self.pretrained = config['pretrained']
        # 得到主干特征提取网络
        self.backbone = getBackbone(config['backbone'], pretrained=self.pretrained)
        self.aux_conv = self._aux_conv(self.aux_conv_in, self.tree_in)
        # 定义树结构
        self.tree = BinaryNeuralTree(class_num=self.class_num, tree_height=self.tree_height, in_channels=self.tree_in)

    def _aux_conv(self, aux_conv_in, aux_conv_out):
        layers = []
        if aux_conv_in > 0:
            extra_conv = nn.Conv2d(in_channels=aux_conv_in, out_channels=aux_conv_out, kernel_size=1)
            extra_conv.apply(weightInit)
            layers.append(extra_conv)
            layers.append(nn.ReLU())
        return nn.Sequential(*layers)

    def forward(self, x):
        # 将输入传入特征提取网络提取特征
        init_feature = self.backbone(x)
        init_feature = self.aux_conv(init_feature)
        # 将特征图传入二叉树结构，得到输出
        leaves_out, final_out, final_features, penultimate_out = self.tree(init_feature)
        # 返回:每个叶子结点的预测概率、倒数第二层结点的预测概率、最终的预测概率、最后一层的特征
        return leaves_out, final_out, final_features, penultimate_out

    二叉树的初始化¶

    class BinaryNeuralTree(nn.Module):
    def __init__(self, class_num, tree_height=3, in_channels=512):
        super(BinaryNeuralTree, self).__init__()

        assert tree_height > 1
        # 类别数
        self.class_num = class_num
        # 树高
        self.tree_height = tree_height
        # 特征图通道数
        self.in_channels = in_channels
        # 计算结点处的routing
        self.branch_routings = self._branch_routings()
        # 计算分支上注意力
        self.attention_transformers = self._attention_transformers()
        # 计算结点的预测概率
        self.label_predictions = self._label_predictions()
        # 计算对数
        self.logs = self._log()

    def _branch_routings(self):
        # structure表示routings模块的结构，structure为二维列表，其中每个元素代表一个结点上的routing，可以通过一个二维坐标来定位该routing位于树的哪个结点位置
        # 树的最后一层不需要routing，因此，如果树高为3的话，structure结构为[1,3](只有前两层包含routing)
        structure = [[None for _ in range(int(pow(2, i)))] for i in range(self.tree_height - 1)]
        cur = 0
        for i in range(self.tree_height - 1):
            for j in range(int(pow(2, i))):
                # 每个结点上的routing均由BranchRoutingModule模块构成
                self.__setattr__('branch_routing_module' + str(cur), BranchRoutingModule(in_channels=self.in_channels))
                structure[i][j] = self.__getattr__('branch_routing_module' + str(cur))
                cur += 1

        return structure

    def _attention_transformers(self):
        # 与_branch_routings函数类似，structure表示attention模块的结构，也是一个二维列表
        # 树的每个分支都需要attention结构，因此如果树高为3，则structure结构可以表示为[2,4]
        structure = [[None for _ in range(int(pow(2, i + 1)))] for i in range(self.tree_height - 1)]
        cur = 0
        for i in range(self.tree_height - 1):
            for j in range(int(pow(2, i + 1))):
                # 需要分奇偶，偶数(右分支)，由一个attention_transformer组成
                if j % 2:# j从0开始，因此偶数代表左侧，奇数代表右侧
                    self.__setattr__('attention_transformer' + str(cur),
                                     nn.Sequential(*[AttentionTransformer(in_channels=self.in_channels)]))
                # 奇数(左分支)，由两个attention_transformer组成
                else:
                    self.__setattr__('attention_transformer' + str(cur),
                                     nn.Sequential(*[AttentionTransformer(in_channels=self.in_channels),
                                                     AttentionTransformer(in_channels=self.in_channels)]))
                structure[i][j] = self.__getattr__('attention_transformer' + str(cur))
                cur += 1

        return structure

    def _label_predictions(self):
        # 表示叶子结点(标签预测模块)的结构，如果高为3，则有4个叶子结点
        structure = [None] * int(pow(2, self.tree_height - 1))
        cur = 0
        for i in range(int(pow(2, self.tree_height - 1))):
            # 每个叶子结点的标签预测均由LabelPredictionModule函数构成
            self.__setattr__('label_prediction_module' + str(cur),
                             LabelPredictionModule(class_num=self.class_num, in_channels=self.in_channels))
            structure[i] = self.__getattr__('label_prediction_module' + str(cur))
            cur += 1

        return structure

    # 用于计算预测结果的对数，为最终计算对数似然损失做准备
    def _log(self, epsilon=1e-12):
        # epsilon为边界值，防止出现预测结果是0的情况(log0无意义)
        structure = [None] * int(pow(2, self.tree_height - 1) + 1)
        cur = 0
        for i in range(int(pow(2, self.tree_height - 1)) + 1):
            # 分别计算输入的对数
            self.__setattr__('log' + str(cur), (lambda x: torch.log(x + epsilon)))
            structure[i] = self.__getattribute__('log' + str(cur))
            cur += 1

        return structure

    二叉树的前向传播¶

    def forward(self, x):
    # probs长度为[2,4]的空列表(二维)
    # 表示每行中，每个结点的路线概率，即论文中的φ，第一行一个结点，有两个路线概率(左和右)，第二行两个，依次类推；
    probs = [[None for _ in range(int(pow(2, i + 1)))] for i in range(self.tree_height - 1)]
    # features长度为[1,2,4]
    # 表示每层中，每个结点的特征，第一层一个，第二层两个，第三层四个，依次类推；
    features = [[None for _ in range(int(pow(2, i)))] for i in range(self.tree_height)]
    # 顶层的特征为特征网络输出的特征，直接赋值即可
    features[0][0] = x
    # 遍历所有的层，从第一层(顶层，只有一个结点)开始
    for i in range(self.tree_height - 1):
        # 在每层中再遍历结点
        for j in range(int(pow(2, i))):
            # 得到当前结点向右偏移的概率，具体计算方法见路线分支模块小节
            temp_prob = self.branch_routings[i][j](features[i][j])
            # 向左便宜概率为1-向右偏移的概率
            probs[i][j * 2] = 1 - temp_prob
            # 向右偏移的概率直接赋值即可
            probs[i][j * 2 + 1] = temp_prob
            # 如果是不是第一层，则本层最终的左右路线概率需要乘上上一层的概率
            if i:
                # 本层左侧最终的路线概率
                probs[i][j * 2] = probs[i][j * 2] * probs[i - 1][j]
                # 本层右侧最终的路线概率
                probs[i][j * 2 + 1] = probs[i][j * 2 + 1] * probs[i - 1][j]
            # 下一层的特征，需要将本层的特征乘上本层路线的attention_transformers模块
            # 不同路线(分左右)对应不同的注意力模块计算方法
            # 先计算当前结点下一层左侧的特征，再计算右侧的特征
            features[i + 1][j * 2] = self.attention_transformers[i][j * 2](features[i][j])
            features[i + 1][j * 2 + 1] = self.attention_transformers[i][j * 2 + 1](features[i][j])
    # 得到每个叶子结点的预测概率
    leaves_out = [self.label_predictions[i](features[self.tree_height - 1][i]) for i in
                  range(int(pow(2, self.tree_height - 1)))]
    # 得到倒数第二层结点特征的预测概率(论文中未提，源码作者新加的:利用倒数第二层结点的预测求对数似然损失以及两层之间的排序损失优化网络)
    # 由于最后一层结点数是倒数第二层结点数的两倍，因此最后一层两个结点(叶子)对应倒数第二层一个结点，因此列索引i要除以2
    penultimate_out = [self.label_predictions[i](features[self.tree_height - 2][i//2]) for i in
                  range(int(pow(2, self.tree_height - 1)))]
    # 所有叶子结点预测概率加权求和，作为最终的类别预测概率
    # 权重为上一层结点的路线选择概率(向左或者向右)
    final_out = probs[self.tree_height - 2][0] * leaves_out[0]
    for i in range(1, int(pow(2, self.tree_height - 1))):
        final_out = final_out + probs[self.tree_height - 2][i] * leaves_out[i]
    # 每个叶子结点特征的预测概率，分别求对数，便于后面计算对数似然损失(下面用途相同)
    leaves_out = tuple((self.logs[i](leaves_out[i]) for i in range(int(pow(2, self.tree_height - 1)))))
    # 倒数第二层结点特征的预测概率，求对数
    penultimate_out = tuple((self.logs[i](penultimate_out[i]) for i in range(int(pow(2, self.tree_height - 1)))))
    # 最终的预测概率，求对数
    final_out = self.logs[int(pow(2, self.tree_height - 1))](final_out)
    # 得到最后一层的特征，即叶子结点的特征，后续计算多样性损失(论文中未提，源码作者新加的)
    final_features = tuple(features[self.tree_height - 1][i] for i in range(int(pow(2, self.tree_height - 1))))
    # 返回:每个叶子结点的预测概率、最终的预测概率、最后一层的特征、倒数第二层结点的预测概率
    return leaves_out, final_out, final_features, penultimate_out

    branch routing module¶

    class BranchRoutingModule(nn.Module):
    def __init__(self, in_channels=512, epsilon=1e-12):
        super(BranchRoutingModule, self).__init__()

        self.conv1 = nn.Conv2d(in_channels, in_channels, kernel_size=1)
        # 定义GlobaL Context Block模块
        self.gcblock = GCBlock(in_channels, in_channels)
        # 定义标准化处理
        self.l2norm = lambda x: F.normalize(x, dim=1)
        # 将所有通道经过线性映射，变成一个数
        self.fc = nn.Linear(in_channels, 1)
        # 全局平均池化
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        # 对应元素开平方根
        self.signedsqrt = lambda x: torch.sign(x) * torch.sqrt(torch.sign(x) * x + epsilon)
        # Sigmoid激活函数，变成[0,1]之间的概率
        self.sigmoid = nn.Sigmoid()

    def forward(self, x):
        feature = self.conv1(x)
        # 经过GlobaL Context Block模块，得到搜集了上下文信息的特征图数据
        feature = self.gcblock(feature)
        # 经过全局平均池化，得到特征向量
        feature = self.avgpool(feature)
        # 将特征数据开根
        feature = self.signedsqrt(feature)
        # 将得到的特征进行标准化操作
        feature = self.l2norm(feature)
        # 将特征数据变为两维
        feature = feature.view(feature.size(0), -1)
        # 输入全连接，进行预测，得到路线分数(选择左还是选择右)
        out = self.fc(feature)
        # 经过sigmoid，将最后的路线分数做归一化处理
        out = self.sigmoid(out)

        return out

    class AttentionTransformer(nn.Module):
    def __init__(self, in_channels=512):
        super(AttentionTransformer, self).__init__()
        # 定义ASPP模块
        self.aspp = ASPP(in_channel=in_channels, depth=in_channels)
        # self.se = SEBlock(in_planes=in_channels // 2, planes=in_channels)

        # self.aspp = nn.Sequential()
        # 定义SE模块
        self.se = SEBlock(in_planes=in_channels, planes=in_channels)

    def forward(self, x):
        # 依次经过aspp、se模块
        feature = self.aspp(x)
        out = self.se(feature)

        return out

    ASPP

    class ASPP(nn.Module):
    def __init__(self, in_channel=512, depth=256):
        super(ASPP, self).__init__()
        # 全局平均池化
        self.mean = nn.AdaptiveAvgPool2d((1, 1))
        self.conv = nn.Conv2d(in_channel, depth, 1, 1)
        # 定义四种扩张速率(dilation)不同的扩张卷积层
        # 扩张速率为1时，卷积核尺寸为1，其余均为3
        self.atrous_block1 = nn.Conv2d(in_channel, depth, kernel_size=1, stride=1)
        self.atrous_block6 = nn.Conv2d(in_channel, depth, kernel_size=3, stride=1, padding=6, dilation=6)
        self.atrous_block12 = nn.Conv2d(in_channel, depth, kernel_size=3, stride=1, padding=12, dilation=12)
        self.atrous_block18 = nn.Conv2d(in_channel, depth, kernel_size=3, stride=1, padding=18, dilation=18)
        # 点积层，用于改变特征图通道数，将堆叠后的特征图通道数压缩成1倍的(与堆叠前相同)
        self.conv_1x1_output = nn.Conv2d(depth * 5, depth, 1, 1)

    def forward(self, x):
        size = x.shape[2:]
        # 首先原图经过全局平均池化操作(求平均值)
        image_features = self.mean(x)
        image_features = self.conv(image_features)
        # 然后放大到与原特征图尺寸一样的大小
        image_features = F.upsample(image_features, size=size, mode='bilinear')
        # 依次经过不同扩展速率的扩张卷积模块，提高特征图感受野
        atrous_block1 = self.atrous_block1(x)
        atrous_block6 = self.atrous_block6(x)
        atrous_block12 = self.atrous_block12(x)
        atrous_block18 = self.atrous_block18(x)
        # 将具有不同感受野的特征图合并，最后经过一层卷积核为1*1的卷积运算
        net = self.conv_1x1_output(torch.cat([image_features, atrous_block1, atrous_block6,
                                              atrous_block12, atrous_block18], dim=1))
        return net

    SEBlock

    class SEBlock(nn.Module):
    def __init__(self, in_planes, planes, stride=1):
        super(SEBlock, self).__init__()
        # 定义一些列卷积操作
        self.conv1 = nn.Conv2d(in_planes, planes, kernel_size=3, stride=stride, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(planes)
        self.conv2 = nn.Conv2d(planes, planes, kernel_size=3, stride=1, padding=1, bias=False)
        self.bn2 = nn.BatchNorm2d(planes)
        # 定义全局平均池化与Sigmoid函数
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.sigmoid = nn.Sigmoid()

        self.shortcut = nn.Sequential()
        if stride != 1 or in_planes != planes:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_planes, planes, kernel_size=1, stride=stride, bias=False),
                nn.BatchNorm2d(planes)
            )

        # SE layers
        # fc1、fc2表示多层感知机(类似于多层线性映射)
        self.fc1 = nn.Conv2d(planes, planes // 16, kernel_size=1)  # Use nn.Conv2d instead of nn.Linear
        self.fc2 = nn.Conv2d(planes // 16, planes, kernel_size=1)

    def forward(self, x):
        # 依次经过卷积、标准化、卷积、标准化
        out = torch.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        # Squeeze，求全局平均值，得到b*1024*1*1的通道权重
        w = self.avgpool(out)
        # 经过一层多层感知机模块
        w = torch.relu(self.fc1(w))
        w = self.sigmoid(self.fc2(w))

        # 将得到的通道权重与原始输出相乘
        out = out * w  # New broadcasting feature from v0.2!

        out += self.shortcut(x)
        out = torch.relu(out)
        return out

    结点预测¶

    class LabelPredictionModule(nn.Module):
    def __init__(self, class_num, in_channels=512):
        super(LabelPredictionModule, self).__init__()
        # 定义好分类数
        self.class_num = class_num
        # 定义标准化与卷积操作
        self.bn = nn.BatchNorm2d(in_channels)
        self.conv1 = nn.Conv2d(in_channels, in_channels, kernel_size=1)
        self.l2norm = lambda x: F.normalize(x, dim=1)
        # 全连接层，用于预测类别
        self.fc = nn.Linear(in_channels, self.class_num, bias=True)
        # 自适应全局平均池化
        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.signedsqrt = lambda x: torch.sign(x) * torch.sqrt(torch.sign(x) * x + 1e-12)
        self.softmax = nn.Softmax(dim=1)

    def forward(self, x):
        # 与一般的特征提取网络类似，依次经过标准化、卷积、全局平均池化，得到用于分类的1024个向量
        feature = self.bn(x)
        feature = self.conv1(feature)
        feature = self.avgpool(feature)
        # 向量中的特征数据开根
        feature = self.signedsqrt(feature)
        # 再经过一次标准化
        feature = self.l2norm(feature)
        # 将特征图数据变成二维的，b*1024
        feature = feature.view(feature.size(0), -1)
        # 经过全连接层得到预测值
        out = self.fc(feature)
        # 再经过softmax，得到预测概率
        out = self.softmax(out)

        return out

    损失函数¶

    class MyLossFunction(object):
    def __init__(self, lambdas, margin):
        self.lambdas = lambdas
        self.margin = margin

    def __call__(self, output, label, target):
        # 网络的返回值:每个叶子结点的预测概率、倒数第二层结点的预测概率、最终的预测概率、最后一层的特征图
        leaves_out, final_out, final_features, penultimate_out = output
        # 各损失的权重系数
        lambda_0, lambda_1, lambda_2, lambda_3, lambda_4 = self.lambdas
        # 排序损失中的边界值
        margin = self.margin
        # 首先计算最终的预测值与标签之间的对数似然损失
        loss = lambda_0 * nn.NLLLoss()(final_out, label)
        # 再分别计算各个叶子的预测值与标签之间的对数似然损失
        for out in leaves_out:
            loss = loss + lambda_1 * nn.NLLLoss()(out, label)

    def train(args, model, optimizers, learning_rate_schedulers, dataloaders):
    loss_records = []
    # 创建模型保存的地址
    if not os.path.exists(args.save_dir):
        os.makedirs(args.save_dir)
    # 创建日志保存的地址
    if args.log_file:
        if not os.path.exists(args.log_dir):
            os.makedirs(args.log_dir)
        with open(os.path.join(args.log_dir, args.log_file), 'a+') as log_file:
            log_file.write(('-' * 5) + '\n')

    train_batch_size = dataloaders['train'].batch_size
    # 每个epoch需要训练多少次
    train_epoch_step = len(dataloaders['train'])

    savepoint = args.savepoint
    checkpoint = args.checkpoint
    if savepoint > train_epoch_step:
        savepoint = train_epoch_step
        checkpoint = savepoint
    # 各个损失的权重，最终的损失由各个损失加权求和而定
    lambdas = (args.lambda_0, args.lambda_1, args.lambda_2, args.lambda_3, args.lambda_4)
    margin = args.margin
    # 定义损失函数
    loss_func = MyLossFunction(lambdas, margin)
    # 初始化两个时间变量(用于计算每次迭代所消耗的时间)
    last_time, cur_time = None, datetime.datetime.now()
    # 第一阶段的训练start_epoch1到epoch_num1，此时更新除了backbone(特征提取网络)之外的网络参数
    for epoch in range(args.start_epoch1, args.epoch_num1):
        # 初始化更新次数
        cur_step = 0
        # 迭代遍历训练集
        for batch_no, data in enumerate(dataloaders['train']):
            # 更新last_time(上一次)
            last_time = cur_time
            # 更新次数加1
            cur_step += 1
            # 模型改为训练模式(参数可更新)
            model.train(True)
            # 将网络中的backbone模块(测试模块)改为测试模式(不更新参数)
            if args.use_cuda:
                model.module.backbone.train(False)
            else:
                model.backbone.train(False)
            # 得到输入与标签
            image, label = data
            # 将张量转化为变量，便于反向传播
            if args.use_cuda:
                image = Variable(image.cuda())
                label = Variable(label.cuda()).long()
            else:
                image = Variable(image)
                label = Variable(label).long()
            # 梯度清零
            optimizers[0].zero_grad()
            # 输入传入模型，得到输出
            leaves_out, final_out, final_features, penultimate_out = model(image)
            # target为了后期计算排序损失，全1表示所有的均是正序(顺序相同)
            target = torch.ones([leaves_out[0].shape[0]])
            # target传入cuda
            if args.use_cuda:
                target = target.cuda()
            # 求总损失
            loss = loss_func((leaves_out, final_out, final_features, penultimate_out), label, target)

            if args.use_cuda:
                # loss传入cuda
                loss = loss.cuda()
            # 反向传播，求梯度
            loss.backward()

            if args.use_cuda:
                # 等待当前设备所有的cuda求完梯度(常用于多卡训练)
                torch.cuda.synchronize()
            # 利用梯度更新参数，利用第一个优化器更新
            optimizers[0].step()

            if args.use_cuda:
                # 与上述类似，等待所有的cuda更新完参数
                torch.cuda.synchronize()
            # 一次更新结束，保存时间
            cur_time = datetime.datetime.now()
            # 储存当次损失
            loss_records.append(loss.detach().item())
            # 输出当前阶段训练信息
            print('train_step: {:-8d} / {:d}, loss: {:6.4f}'
                  .format(cur_step, train_epoch_step, loss.detach().item()), flush=True)

            print(cur_time - last_time)
        # 更新第一个优化器的学习率
        if learning_rate_schedulers[0]:
            learning_rate_schedulers[0].step()

        print('stage 1')
        print('epoch: {:-4d}, start_epoch: {:-4d}, epoch_num: {:-4d}.'
              .format(epoch, args.start_epoch1, args.epoch_num1))
        # 每次训练完是否验证
        if args.type == 'val':
            if (epoch + 1) % args.val_interval == 0:
                # 传入test函数，测试模型
                val_result = test(args, model=model, dataloader=dataloaders['val'], type='val')
                # 得到验证精度
                val_acc = val_result['val_acc']
                # 输出验证精度
                val_acc_str = 'val_acc: {:6.4f}.'.format(val_acc)
                print(val_acc_str)
                # 保存日志
                if args.log_file:
                    with open(os.path.join(args.log_dir, args.log_file), 'a+') as log_file:
                        log_file.write('stage 1 ' + 'epoch ' + str(epoch) + ', ' + val_acc_str + '\n')
        # 与前面类似，等待所有cuda上的运算结束，再执行下一步
        if args.use_cuda:
            torch.cuda.synchronize()
        # 保存日志
        if (epoch + 1) % args.save_interval == 0:
            save_path = os.path.join(args.save_dir, 'weight_stage' + str(1) + '_epoch' + str(epoch) + '.pth')
            torch.save(model.state_dict(), save_path)
        # 自动清理显存
        if args.use_cuda:
            torch.cuda.empty_cache()

    def test(args, model, dataloader, type):
    assert type in ['val', 'test']
    # 初始化结果字典
    result = {}
    # 损失变化过程
    loss_records = []

    batch_size = dataloader.batch_size
    # 一共的迭代次数
    epoch_step = len(dataloader)
    # 测试集总数
    data_num = len(dataloader.dataset)
    # 计算准确预测的图片个数
    correct = 0
    # 类别数量
    class_num = args.class_num
    # 混淆矩阵
    confusion_matrix = [[0] * class_num for _ in range(class_num)]
    # 损失权重系数
    lambdas = (args.lambda_0, args.lambda_1, args.lambda_2, args.lambda_3, args.lambda_4)
    # 排序损失的边界值
    margin = args.margin
    # 用于计算总损失
    loss_func = MyLossFunction(lambdas, margin)
    # 模型转化为测试状态(不更新模型参数)
    model.train(False)

    with torch.no_grad():
        # 与训练过程类似，初始化迭代次数
        cur_step = 0
        # 迭代读取数据集
        for batch_no, data in enumerate(dataloader):

            cur_step += 1
            # 得到测试数据与标签
            image, label = data
            # 显卡
            if args.use_cuda:
                image = Variable(image.cuda())
                label = Variable(label.cuda()).long()
            else:
                image = Variable(image)
                label = Variable(label).long()
            # 输入传入模型中，得到输出
            leaves_out, final_out, final_features, penultimate_out = model(image)
            # 与训练过程类似，target用作排序损失中
            target = torch.ones([leaves_out[0].shape[0]])

            if args.use_cuda:
                target = target.cuda()
            # 得到测试损失
            loss = loss_func((leaves_out, final_out, final_features, penultimate_out), label, target)

            if args.use_cuda:
                loss = loss.cuda()
            # 每次迭代得到的损失都保存起来
            loss_records.append(loss.detach().item())
            # 输出当前测试过程中，迭代数据
            print(type + '_step: {:-8d} / {:d}, loss: {:6.4f}.'
                  .format(cur_step, epoch_step, loss.detach().item()), flush=True)
            # 得到预测结果，计算正确预测的个数
            _, pred = torch.topk(final_out, 1)
            correct += torch.sum((pred[:, 0] == label)).data.item()
            # 混淆矩阵
            for i in range(label.shape[0]):
                confusion_matrix[label[i]][pred[i, 0]] += 1
    # 正确数除以测试集总数得到准确率
    acc = correct / data_num
    # 损失变化过程
    result['loss_records'] = loss_records
    # 测试精度
    result[type + '_acc'] = acc
    # 混淆矩阵
    result[type + '_cfs_mat'] = confusion_matrix

    return result
