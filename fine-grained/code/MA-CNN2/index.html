    综述¶

    会议时间：IEEE International Conference on Computer Vision 2017 (ICCV 2017)

    论文地址：http://openaccess.thecvf.com/content_ICCV_2017/papers/Zheng_Learning_Multi-Attention_Convolutional_ICCV_2017_paper.pdf

    源码地址：https://github.com/Jianlong-Fu/Multi-Attention-CNN（官方代码，caffe版本）、https://github.com/liangnjupt/Multi-Attention-CNN-pytorch（PyTorch版本）

    整体结构¶

    class MACNN(nn.Module):
    # 参数初始化
    def __init__(self, class_num):
        super(MACNN, self).__init__()
        self.vgg = vgg.vgg19(True)
        self.feat_dims = 512
        self.class_num = class_num
        # 定义四个通道分组层，即论文中的channel grouping layers
        self.se1 = SELayer(self.feat_dims)
        self.se2 = SELayer(self.feat_dims)
        self.se3 = SELayer(self.feat_dims)
        self.se4 = SELayer(self.feat_dims)
        # 定义全局平均池化层
        self.pool = nn.AdaptiveAvgPool2d(1)
        # 定义全连接层，用于对图像做分类
        self.fc1 = nn.Linear(self.feat_dims, self.class_num)
        self.fc2 = nn.Linear(self.feat_dims, self.class_num)
        self.fc3 = nn.Linear(self.feat_dims, self.class_num)
        self.fc4 = nn.Linear(self.feat_dims, self.class_num)
        self.fcall = nn.Linear(5 * self.feat_dims, self.class_num)

    # 前向传播
    def forward(self, x):
        # 先将图片传入特征提取网络提取特征
        feat_maps = self.vgg(x)
        # 依次将得到的特征传入通道分组层，得到局部特征(Pi)和局部注意力图(Mi)
        # 其中局部特征Pi是由特征x和局部注意力图Mi相乘，再全局池化得到
        P1, M1 = self.se1(feat_maps)
        P2, M2 = self.se2(feat_maps)
        P3, M3 = self.se3(feat_maps)
        P4, M4 = self.se4(feat_maps)
        # 依次将局部特征传入定义好的全连接层，得到每个部位的分类结果
        pred1 = self.fc1(P1.flatten(1))
        pred2 = self.fc2(P2.flatten(1))
        pred3 = self.fc3(P3.flatten(1))
        pred4 = self.fc4(P4.flatten(1))
        # 将局部特征与全局特征合并，再传入全连接层得到最终的分类结果
        P = torch.cat([P1, P2, P3, P4, self.pool(feat_maps)], dim=1)
        pred = self.fcall(P.flatten(1))

        return [P1, P2, P3, P4], \
            [M1, M2, M3, M4], \
            [pred1, pred2, pred3, pred4, pred]

    通道分组层¶

    class SELayer(nn.Module):
    def __init__(self, channel, reduction=1):
        super(SELayer, self).__init__()
        self.avg_pool = nn.AdaptiveAvgPool2d(1)
        # 定义全连接层，对应论文公式3中的fi(·)
        self.fc = nn.Sequential(
            nn.Linear(channel, channel // reduction, bias=False),
            nn.Tanh(),
            nn.Linear(channel // reduction, channel, bias=False)
        )

    def forward(self, x):
        b, c, h, w = x.size()
        # 先将特征依次传入全局池化、全连接层，对应论文公式3
        y = self.avg_pool(x).view(b, c)
        y = self.fc(y)
        y_ = y.view(b, c, 1, 1)

        # paper revised version
        # 沿通道求和,之后再将注意力图做sigmoid归一化处理，得到局部注意力图，对应论文公式4
        M = torch.sum(x * y_.expand_as(x), dim=1, keepdim=True)  # b,1,h,w
        M = torch.sigmoid(M).view(b, 1, h, w)
        # 将注意力图和图片特征做点乘，得到局部特征
        P = x * M.expand_as(x)  # b,1,h,w
        P = F.avg_pool2d(P, (h, w))  # b,c,1,1

        return P, M.squeeze(dim=1)

    距离损失¶

    用于鼓励同一分支的注意力去关注物体相同的部位

    # 距离损失
class DisLoss(nn.Module):
    def __init__(self):
        super(DisLoss, self).__init__()
        return

    def forward(self, x):
        '''
        :param x:  b,h,w
        :return:
        '''
        b, h, w = x.shape
        x = x.view(b, -1)
        # 定位注意力图最大值所在的位置
        num = torch.argmax(x, dim=1)
        # 得到最值处的横纵坐标
        cx = num % h
        cy = num // h
        # 距离图，也就是一张h×w的图，图上的点距离最值点越远，值越大
        # 对应论文公式8中后半部分(算距离)
        maps = self.get_maps(h, cx, cy)
        maps = torch.from_numpy(maps).to(x.device)
        # 距离和注意力数据点乘求和，得到损失
        # 该损失用于鼓励每个分支的注意力数据都集中于一个位置，即集中在最大值处
        part = x * maps
        loss = torch.sum(part) / b
        return loss

    # 计算最值距离
    def get_maps(self, a, cx, cy):
        batch_size = len(cx)
        cx = cx.data.cpu().numpy()
        cy = cy.data.cpu().numpy()
        maps = np.zeros((batch_size, a * a), dtype=np.float32)
        # 先得到两组连续数据，分别表示注意力图的长宽
        rows = np.arange(a)
        cols = np.arange(a)
        coords = np.empty((len(rows), len(cols), 2), dtype=np.intp)
        # 两组数据组成坐标矩阵，这里coords尺寸为(h*w, 2)，第二维度储存x，y坐标数据
        coords[..., 0] = rows[:, None]
        coords[..., 1] = cols
        coords = coords.reshape(-1, 2)
        # 逐batch遍历
        for b in range(batch_size):
            vec = np.array([cy[b], cx[b]])
            # 计算当前坐标点与最值处坐标点的欧氏距离，得到距离数据，存入maps中
            maps[b, :] = np.linalg.norm(coords - vec, axis=1)
        return maps

    多样性损失¶

    用于鼓励网络不同注意力分支去关注物体不同部位

    # 多样性损失
class DivLoss(nn.Module):
    def __init__(self):
        super(DivLoss, self).__init__()
        return

    def forward(self, x):
        '''
        :param x: [b,h,w]
        :return:
        '''
        # 依次提取四个分支的局部注意力图，对应论文Mi
        x1 = x[0]
        x2 = x[1]
        x3 = x[2]
        x4 = x[3]
        # 对应论文公式9中的边缘参数mrg
        mrg = 0.02
        # 将二维数据压缩为一维数据，便于逐元素运算
        # 这里xi对应论文公式9中的mi(x, y)
        x1 = x1.view(x1.size(0), -1)
        x2 = x2.view(x2.size(0), -1)
        x3 = x3.view(x3.size(0), -1)
        x4 = x4.view(x4.size(0), -1)
        # 合并mk(x, y)数据，便于之后再比出最大值
        tmp1 = torch.stack((x2, x3, x4))
        tmp2 = torch.stack((x1, x3, x4))
        tmp3 = torch.stack((x1, x2, x4))
        tmp4 = torch.stack((x1, x2, x3))
        # 沿分支方向比出最大值，也就是逐点比出三个分支的最大响应
        t1, _ = torch.max(tmp1, dim=0)
        t2, _ = torch.max(tmp2, dim=0)
        t3, _ = torch.max(tmp3, dim=0)
        t4, _ = torch.max(tmp4, dim=0)
        # 最大响应减去边缘参数，提升模型鲁棒性
        t1 = t1 - mrg
        t2 = t2 - mrg
        t3 = t3 - mrg
        t4 = t4 - mrg
        # 逐元素点乘求和，得到距离损失，对应论文公式9
        loss = (torch.sum(x1 * t1) + torch.sum(x2 * t2) + torch.sum(x3 * t3) + torch.sum(x4 * t4)) / x1.size(0)
        return loss
