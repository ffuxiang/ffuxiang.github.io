    def __init__(self, model, feature_size, classes_num):
    super(PMG, self).__init__()
    # 定义特征提取网络
    self.features = model
    # 分别给三个阶段定义三种最大池化运算
    self.max1 = nn.MaxPool2d(kernel_size=56, stride=56)
    self.max2 = nn.MaxPool2d(kernel_size=28, stride=28)
    self.max3 = nn.MaxPool2d(kernel_size=14, stride=14)
    # 特征提取网络最后输出的特征数量，即全连接层的输入尺寸
    self.num_ftrs = 2048 * 1 * 1
    # ELU激活函数
    self.elu = nn.ELU(inplace=True)
    # 三个特征图融合阶段的全连接层
    # 依次经过批量归一化、第一个线性回归、批量归一化、ELU激活函数、第二个线性回归
    # 融合后的特征图通道数是前三个阶段特征图通道数的三倍，因此要乘以3
    self.classifier_concat = nn.Sequential(
        nn.BatchNorm1d(1024 * 3),
        nn.Linear(1024 * 3, feature_size),
        nn.BatchNorm1d(feature_size),
        nn.ELU(inplace=True),
        nn.Linear(feature_size, classes_num),
    )
    # 第一阶段的Conv Block模块，由两个BasicConv构成，特征图数从512(第一阶段的特征图通道数)到512再到1024
    self.conv_block1 = nn.Sequential(
        BasicConv(self.num_ftrs//4, feature_size, kernel_size=1, stride=1, padding=0, relu=True),
        BasicConv(feature_size, self.num_ftrs//2, kernel_size=3, stride=1, padding=1, relu=True)
    )
    # 第一阶段的分类器，与上述融合后的特征图经过的分类器类似
    # 由批量归一化、第一个线性回归、批量归一化、ELU激活函数、第二个线性回归五个模块组成
    self.classifier1 = nn.Sequential(
        nn.BatchNorm1d(self.num_ftrs//2),
        nn.Linear(self.num_ftrs//2, feature_size),
        nn.BatchNorm1d(feature_size),
        nn.ELU(inplace=True),
        nn.Linear(feature_size, classes_num),
    )
    # 第二阶段的Conv Block模块，由两个BasicConv构成，特征图数从1024(第二阶段的特征图通道数)到512再到1024
    self.conv_block2 = nn.Sequential(
        BasicConv(self.num_ftrs//2, feature_size, kernel_size=1, stride=1, padding=0, relu=True),
        BasicConv(feature_size, self.num_ftrs//2, kernel_size=3, stride=1, padding=1, relu=True)
    )
    # 第二阶段的分类器，与上述分类器构成类似
    self.classifier2 = nn.Sequential(
        nn.BatchNorm1d(self.num_ftrs//2),
        nn.Linear(self.num_ftrs//2, feature_size),
        nn.BatchNorm1d(feature_size),
        nn.ELU(inplace=True),
        nn.Linear(feature_size, classes_num),
    )
    # 第三阶段的Conv Block模块，由两个BasicConv构成，特征图数从2048(第三阶段的特征图通道数)到512再到1024
    self.conv_block3 = nn.Sequential(
        BasicConv(self.num_ftrs, feature_size, kernel_size=1, stride=1, padding=0, relu=True),
        BasicConv(feature_size, self.num_ftrs//2, kernel_size=3, stride=1, padding=1, relu=True)
    )
    # 第三阶段的分类器，与上述分类器构成类似
    self.classifier3 = nn.Sequential(
        nn.BatchNorm1d(self.num_ftrs//2),
        nn.Linear(self.num_ftrs//2, feature_size),
        nn.BatchNorm1d(feature_size),
        nn.ELU(inplace=True),
        nn.Linear(feature_size, classes_num),
    )

    前向传播¶

    def forward(self, x):
    # 首先，将原始数据经过特征提取，得到特征提取不同阶段的特征图
    # 后面只用到了xf3, xf4, xf5三组特征图，对应前三个阶段
    xf1, xf2, xf3, xf4, xf5 = self.features(x)
    # 然后三组特征图依次经过预先定义好的Conv Block模块
    xl1 = self.conv_block1(xf3)
    xl2 = self.conv_block2(xf4)
    xl3 = self.conv_block3(xf5)
    # 下面对每个阶段的图做分类预测
    # 第一阶段，首先经过最大池化，然后经过分类器进行分类预测
    xl1 = self.max1(xl1)
    xl1 = xl1.view(xl1.size(0), -1)
    xc1 = self.classifier1(xl1)
    # 第二阶段，过程与第一阶段类似
    xl2 = self.max2(xl2)
    xl2 = xl2.view(xl2.size(0), -1)
    xc2 = self.classifier2(xl2)
    # 第三阶段，过程同样与第一阶段类似
    xl3 = self.max3(xl3)
    xl3 = xl3.view(xl3.size(0), -1)
    xc3 = self.classifier3(xl3)
    # 将三个阶段得到的特征图做堆叠，合并到一块，作为第四阶段的特征图
    x_concat = torch.cat((xl1, xl2, xl3), -1)
    # 融合后的特征图，经过分类器做分类预测
    x_concat = self.classifier_concat(x_concat)
    # 依次返回四个阶段的预测概率
    return xc1, xc2, xc3, x_concat

    def forward(self, x):
    x = self.conv1(x)
    x = self.bn1(x)
    x = self.relu(x)
    # x1、x2、x3、x4、x5分别对应不同阶段的特征图
    # 阶段由下采样池化操作划分
    # 每经过一次下采样，就代表从一个阶段过渡到另一个阶段
    x1 = self.maxpool(x)

    x2 = self.layer1(x1)
    x3 = self.layer2(x2)
    x4 = self.layer3(x3)
    x5 = self.layer4(x4)

    x = self.avgpool(x5)
    x = x.reshape(x.size(0), -1)
    x = self.fc(x)
    # 最后返回五个阶段的特征图
    return x1, x2, x3, x4, x5

    卷积模块¶

    class BasicConv(nn.Module):
    # 用于构成论文中的Conv Block模块，依次经过卷积、批量归一化、relu
    # 类似resnet中的一个小模块
    def __init__(self, in_planes, out_planes, kernel_size, stride=1, padding=0, dilation=1, groups=1, relu=True, bn=True, bias=False):
        super(BasicConv, self).__init__()
        self.out_channels = out_planes
        self.conv = nn.Conv2d(in_planes, out_planes, kernel_size=kernel_size,
                              stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)
        self.bn = nn.BatchNorm2d(out_planes, eps=1e-5,
                                 momentum=0.01, affine=True) if bn else None
        self.relu = nn.ReLU() if relu else None

    def forward(self, x):
        x = self.conv(x)
        if self.bn is not None:
            x = self.bn(x)
        if self.relu is not None:
            x = self.relu(x)
        return x

    def train(nb_epoch, batch_size, store_name, resume=False, start_epoch=0, model_path=None):
    # setup output
    exp_dir = store_name
    try:
        os.stat(exp_dir)
    except:
        os.makedirs(exp_dir)

    use_cuda = torch.cuda.is_available()
    print(use_cuda)

    # Data
    # 定义数据集的预处理过程
    print('==> Preparing data..')
    transform_train = transforms.Compose([
        transforms.Scale((550, 550)),
        transforms.RandomCrop(448, padding=8),
        transforms.RandomHorizontalFlip(),
        transforms.ToTensor(),
        transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5)),
    ])
    # 加载数据集
    trainset = torchvision.datasets.ImageFolder(root='./bird/train', transform=transform_train)
    trainloader = torch.utils.data.DataLoader(trainset, batch_size=batch_size, shuffle=True, num_workers=4)

    # Model 如果存在预训练模型，就加载，否则初始化一个模型
    if resume:
        net = torch.load(model_path)
    else:
        net = load_model(model_name='resnet50_pmg', pretrain=True, require_grad=True)
    # 并行训练，将模型同时放入编号为0,1的显卡
    netp = torch.nn.DataParallel(net, device_ids=[0,1])

    # GPU
    device = torch.device("cuda:0,1")
    net.to(device)
    # cudnn.benchmark = True
    # 定义损失函数
    CELoss = nn.CrossEntropyLoss()
    # 定义优化器，分别以不同的学习率优化不同的模型参数
    optimizer = optim.SGD([
        {'params': net.classifier_concat.parameters(), 'lr': 0.002},
        {'params': net.conv_block1.parameters(), 'lr': 0.002},
        {'params': net.classifier1.parameters(), 'lr': 0.002},
        {'params': net.conv_block2.parameters(), 'lr': 0.002},
        {'params': net.classifier2.parameters(), 'lr': 0.002},
        {'params': net.conv_block3.parameters(), 'lr': 0.002},
        {'params': net.classifier3.parameters(), 'lr': 0.002},
        {'params': net.features.parameters(), 'lr': 0.0002}

    ],
        momentum=0.9, weight_decay=5e-4)

    max_val_acc = 0
    # 默认的学习率
    lr = [0.002, 0.002, 0.002, 0.002, 0.002, 0.002, 0.002, 0.0002]

    训练过程¶

    for epoch in range(start_epoch, nb_epoch):
    print('\nEpoch: %d' % epoch)
    net.train()
    # 初始化所有的损失等参数
    train_loss = 0
    train_loss1 = 0
    train_loss2 = 0
    train_loss3 = 0
    train_loss4 = 0
    correct = 0
    total = 0
    idx = 0
    for batch_idx, (inputs, targets) in enumerate(trainloader):
        idx = batch_idx
        if inputs.shape[0] < batch_size:
            continue
        if use_cuda:
            # 是否使用cuda
            inputs, targets = inputs.to(device), targets.to(device)
        inputs, targets = Variable(inputs), Variable(targets)

        # update learning rate
        # 学习率更新策略
        for nlr in range(len(optimizer.param_groups)):
            # 调整学习率，具体调整策略见cosine_anneal_schedule
            optimizer.param_groups[nlr]['lr'] = cosine_anneal_schedule(epoch, nb_epoch, lr[nlr])

        # Step 1
        optimizer.zero_grad()
        # 将原图打乱，每行每列分别划分8份，一共64个
        inputs1 = jigsaw_generator(inputs, 8)
        # 网络第一层的预测概率
        output_1, _, _, _ = netp(inputs1)
        # 第一阶段预测概率与标签做交叉熵损失
        loss1 = CELoss(output_1, targets) * 1
        # 反向传播、更新梯度
        loss1.backward()
        optimizer.step()

        # 下述第2、3阶段的训练过程与第1阶段类似
        # Step 2
        optimizer.zero_grad()
        # 打乱原图，每行每列分别划分4份
        inputs2 = jigsaw_generator(inputs, 4)
        # 得到网络第二层的预测概率
        _, output_2, _, _ = netp(inputs2)
        # 求损失、反向传播、更新参数
        loss2 = CELoss(output_2, targets) * 1
        loss2.backward()
        optimizer.step()

        # Step 3
        optimizer.zero_grad()
        # 同上打乱原图、每行每列划分两份
        inputs3 = jigsaw_generator(inputs, 2)
        # 得到网络第三层的预测概率(即最后一层)
        _, _, output_3, _ = netp(inputs3)
        # 求损失、反向传播、更新参数
        loss3 = CELoss(output_3, targets) * 1
        loss3.backward()
        optimizer.step()

        # Step 4 第四阶段
        optimizer.zero_grad()
        # 输入原图，不进行打乱，得到三个阶段融合后特征图的预测概率
        _, _, _, output_concat = netp(inputs)
        # 求损失、反向传播、更新梯度
        concat_loss = CELoss(output_concat, targets) * 2
        concat_loss.backward()
        optimizer.step()

        #  training log
        # 利用output_concat计算模型的预测精度
        _, predicted = torch.max(output_concat.data, 1)
        # total代表图片总数
        total += targets.size(0)
        # 计算预测正确个数correct
        correct += predicted.eq(targets.data).cpu().sum()
        # 计算训练总损失：总损失以及各阶段损失
        train_loss += (loss1.item() + loss2.item() + loss3.item() + concat_loss.item())
        train_loss1 += loss1.item()
        train_loss2 += loss2.item()
        train_loss3 += loss3.item()
        train_loss4 += concat_loss.item()
        # batch_idx代表训练次数，每50次输出一次信息
        if batch_idx % 50 == 0:
            print(
                'Step: %d | Loss1: %.3f | Loss2: %.5f | Loss3: %.5f | Loss_concat: %.5f | Loss: %.3f | Acc: %.3f%% (%d/%d)' % (
                batch_idx, train_loss1 / (batch_idx + 1), train_loss2 / (batch_idx + 1),
                train_loss3 / (batch_idx + 1), train_loss4 / (batch_idx + 1), train_loss / (batch_idx + 1),
                100. * float(correct) / total, correct, total))
    # 计算训练平均精度以及平均损失
    train_acc = 100. * float(correct) / total
    train_loss = train_loss / (idx + 1)
    # 将训练结果写入results_train.txt文件，类似输出日志
    with open(exp_dir + '/results_train.txt', 'a') as file:
        file.write(
            'Iteration %d | train_acc = %.5f | train_loss = %.5f | Loss1: %.3f | Loss2: %.5f | Loss3: %.5f | Loss_concat: %.5f |\n' % (
            epoch, train_acc, train_loss, train_loss1 / (idx + 1), train_loss2 / (idx + 1), train_loss3 / (idx + 1),
            train_loss4 / (idx + 1)))
    # 测试以及模型保存策略
    # 这里是源码默认的策略：迭代次数小于5次或大于等于80次进行测试
    # 可以通过修改这里，来修改测试以及模型保存策略
    if epoch < 5 or epoch >= 80:
        # 测试得到两种测试精度，对应论文中公式(4)(5)两种计算方法
        # 一个是只用融合特征图得到的预测概率，一个是综合所有的预测概率
        # 并且得到测试损失
        val_acc, val_acc_com, val_loss = test(net, CELoss, 3)
        # 如果当前测试精度大于max，就保存一次模型
        if val_acc_com > max_val_acc:
            # 更新当前最大的精度
            max_val_acc = val_acc_com
            net.cpu()
            # 只保存模型，可以通过修改这里，来修改保存内容
            torch.save(net, './' + store_name + '/model.pth')
            net.to(device)
        # 输出日志
        with open(exp_dir + '/results_test.txt', 'a') as file:
            file.write('Iteration %d, test_acc = %.5f, test_acc_combined = %.5f, test_loss = %.6f\n' % (
            epoch, val_acc, val_acc_com, val_loss))
    else:
        # 如果不测试，就每训练一次，保存一次模型
        net.cpu()
        torch.save(net, './' + store_name + '/model.pth')
        net.to(device)

    拼图的生成¶

    def jigsaw_generator(images, n):
    # n代表每行、每列被分割的份数，必须可以整除原图像的长宽，一共被分割成n^2个小区域
    # l代表区域的横纵位置，首先定义，然后进行打乱
    l = []
    for a in range(n):
        for b in range(n):
            l.append([a, b])
    # 每个区域的尺寸
    block_size = 448 // n
    # 一共得到n^2个区域
    rounds = n ** 2
    # 将l中的元素随机打乱
    random.shuffle(l)
    # 复制一份原图像，防止改变jigsaws的同时改变images(视图)
    jigsaws = images.clone()
    # 打乱后的区域为顺序，做遍历
    for i in range(rounds):
        # 得到区域的横纵位置
        x, y = l[i]
        # 储存左上角的区域，便于后续的使用
        temp = jigsaws[..., 0:block_size, 0:block_size].clone()
        # 将左上角区域的数据赋值为指定区域的数据
        jigsaws[..., 0:block_size, 0:block_size] = jigsaws[..., x * block_size:(x + 1) * block_size,
                                                y * block_size:(y + 1) * block_size].clone()
        # 再将指定区域的数据赋值为原左上角区域的数据
        jigsaws[..., x * block_size:(x + 1) * block_size, y * block_size:(y + 1) * block_size] = temp
        # 上述过程相当于，将左上角的区域和指定的区域(l[i]定位的区域)做交换，从而达到打乱原图的效果
    return jigsaws

    # 定义测试过程中数据的预处理
    transform_test = transforms.Compose([
        transforms.Scale((550, 550)),
        transforms.CenterCrop(448),
        transforms.ToTensor(),
        transforms.Normalize((0.5, 0.5, 0.5), (0.5, 0.5, 0.5)),
    ])
    # 加载测试数据
    testset = torchvision.datasets.ImageFolder(root='./bird/test',
                                               transform=transform_test)
    testloader = torch.utils.data.DataLoader(testset, batch_size=batch_size, shuffle=True, num_workers=4)
    # 遍历数据集
    for batch_idx, (inputs, targets) in enumerate(testloader):
        idx = batch_idx
        # 是否利用cuda
        if use_cuda:
            inputs, targets = inputs.to(device), targets.to(device)
        inputs, targets = Variable(inputs, volatile=True), Variable(targets)
        # 将输入图片直接传入网络，得到四组预测概率
        # 注意、测试阶段无需经过拼图操作
        output_1, output_2, output_3, output_concat= net(inputs)
        # 计算C2预测概率，即论文中的公式(5)，综合了所有的预测概率
        outputs_com = output_1 + output_2 + output_3 + output_concat
        # 计算C1预测概率(只利用融合热图进行分类预测)的损失
        loss = criterion(output_concat, targets)
        # 计算测试损失
        test_loss += loss.item()
        # 得到output_concat(C1)预测的标签
        _, predicted = torch.max(output_concat.data, 1)
        # 得到C2预测的标签
        _, predicted_com = torch.max(outputs_com.data, 1)
        # total表示测试集图片总数
        total += targets.size(0)
        # 依次计算，得到C1准确预测的样本个数，以及C2准确预测的样本个数
        correct += predicted.eq(targets.data).cpu().sum()
        correct_com += predicted_com.eq(targets.data).cpu().sum()
        # 每测试50组，输出一次信息
        if batch_idx % 50 == 0:
            print('Step: %d | Loss: %.3f | Acc: %.3f%% (%d/%d) |Combined Acc: %.3f%% (%d/%d)' % (
            batch_idx, test_loss / (batch_idx + 1), 100. * float(correct) / total, correct, total, 100. * float(correct_com) / total, correct_com, total))
    # 测试集遍历完之后，分别计算得到C1准确率，C2准确率、测试损失
    test_acc = 100. * float(correct) / total
    test_acc_en = 100. * float(correct_com) / total
    test_loss = test_loss / (idx + 1)
    # 将上述三个指标返回
    return test_acc, test_acc_en, test_loss
