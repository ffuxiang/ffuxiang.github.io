    RA-CNN网络整体结构¶

    初始化阶段¶

    def __init__(self, num_classes, pretrained = True):
    super(RACNN, self).__init__()
    # 判断特征提取网络是否使用预训练参数
    if pretrained:
        pretrained = True
    else:
        pretrained = False
    # 定义好三个尺度的特征提取层，分别进行不同的特征提取操作
    self.b1 = vgg19_bn(num_classes = 1000, pretrained = pretrained)
    self.b2 = vgg19_bn(num_classes = 1000, pretrained = pretrained)
    self.b3 = vgg19_bn(num_classes = 1000, pretrained = pretrained)
    # 定义好两个池化层，用于特征提取后第1条分支(分类)的全局池化操作
    # 由于第一尺度输入尺寸为448*448，二三尺度输入尺寸为224*224
    # 两者输入尺寸不同因此需要定义两种池化网络
    # 第一尺度的图像经过特征提取后得到28*28的特征图
    self.feature_pool1 = nn.AvgPool2d(kernel_size = 28, stride = 28)
    # 二三尺度的图像经过特征提取后得到14*14的特征图
    self.feature_pool2 = nn.AvgPool2d(kernel_size = 14, stride = 14)
    # 由于第一尺度得到28*28的图像，因此需要经过降维
    # 变为14*14的图像，再生成注意力区域
    self.atten_pool = nn.MaxPool2d(kernel_size = 2, stride = 2)
    # 生成用于定位注意力区域的坐标(三个数值)
    # 第一尺度到第二尺度
    self.apn1 = nn.Sequential(
        nn.Linear(512 * 14 * 14, 1024),
        nn.Tanh(),
        nn.Linear(1024, 3),#最终得到三个数
        nn.Sigmoid(),
    )
    # 第二尺度到第三尺度
    self.apn2 = nn.Sequential(
        nn.Linear(512 * 14 * 14, 1024),
        nn.Tanh(),
        nn.Linear(1024, 3),
        nn.Sigmoid(),
    )
    # 用于裁剪操作，AttentionCropLayer的定义见下文
    self.crop_resize = AttentionCropLayer()
    # 各个尺度的第一条分支，即经过全连接层进行分类
    self.classifier1 = nn.Linear(512, num_classes)
    self.classifier2 = nn.Linear(512, num_classes)
    self.classifier3 = nn.Linear(512, num_classes)

    前向传播过程¶

    def forward(self, x):
    # 首先将图片输入到第一尺度的特征提取网络中，进行特征提取操作
    conv5_4 = self.b1.features[:-1](x)
    # 第一条分支，将特征图进行全局池化操作，用于后期输入到全连接层
    pool5 = self.feature_pool1(conv5_4)
    # 第二条分支，将特征图先经过池化，再输入到定义好的APN1模块
    # 生成用于定位的注意力区域坐标
    atten1 = self.apn1(self.atten_pool(conv5_4).view(-1, 512 * 14 * 14))
    # 由于坐标经过了归一化操作，因此需要乘上原图大小(448)复原
    # 再将原图像与坐标输入到定义好的裁剪函数里，裁剪生成用于第二尺度学习/预测的注意力区域(图像2)
    scaledA_x = self.crop_resize(x, atten1 * 448)

    # 第一尺度大部分工作就结束了，接下来是第二尺度的工作
    # 第二尺度与第一尺度类似，首先将图像2进行特征提取
    conv5_4_A = self.b2.features[:-1](scaledA_x)
    # 之后将特征提取后的图像放入全局池化层(第一分支)
    pool5_A = self.feature_pool2(conv5_4_A)
    # 将特征提取后的图像放入APN2函数，再生成注意力区域
    atten2 = self.apn2(conv5_4_A.view(-1, 512 * 14 * 14))
    # 再将图像2进行裁剪，生成用于第三尺度学习/预测的注意力区域(图像3)
    scaledAA_x = self.crop_resize(scaledA_x, atten2 * 224)

    # 第三尺度只有特征提取操作，将图像3放入特征提取层，生成特征图
    # 再将特征图经过一层全局池化操作，用于输入到后面的全连接层
    pool5_AA = self.feature_pool2(self.b3.features[:-1](scaledAA_x))

    # 将三个尺度全局池化得到的数据变形，变为N行512列的数组
    pool5 = pool5.view(-1, 512)
    pool5_A = pool5_A.view(-1, 512)
    pool5_AA = pool5_AA.view(-1, 512)
    # 最后将数组传入全连接层进行分类预测
    logits1 = self.classifier1(pool5)
    logits2 = self.classifier2(pool5_A)
    logits3 = self.classifier3(pool5_AA)
    # 返回三个尺度的预测概率、特征图数据、注意力建议数据(用于定位的坐标)、经过裁剪得到的注意力区域
    return [logits1, logits2, logits3], [conv5_4, conv5_4_A], [atten1, atten2], [scaledA_x, scaledAA_x]

    在返回的数据中，三个尺度的预测概率用于最终的物体种类预测、特征图数据与注意力建议数据用于APN参数的预训练，经过裁剪得到的注意力区域可用于最后的验证。

    AttentionCropFunction裁剪函数¶

    前向传播过程

    def forward(self, images, locs):
    # 提前构建好h函数，用于得到裁剪掩模
    h = lambda x: 1. / (1. + torch.exp(-10. * x))
    in_size = images.size()[2]
    unit = torch.stack([torch.arange(0, in_size)] * in_size).float()
    x = torch.stack([unit.t()] * 3)
    y = torch.stack([unit] * 3)
    if isinstance(images, torch.cuda.FloatTensor):
        # isinstance() 函数来判断一个对象是否是一个已知的类型,判断images是否是cuda.FloatTensor类型
        x, y = x.cuda(), y.cuda()

    in_size = images.size()[2]
    ret = []
    # 按batch_size遍历
    for i in range(images.size(0)):
        tx, ty, tl = locs[i][0], locs[i][1], locs[i][2]
        tl = tl if tl > (in_size/3) else in_size/3
        # 对窗口长度进行修正，限制tl不能过小，最小是in_size/3
        # 如果距离过小的话，会导致裁剪得到的窗口很小，容易丢失信息
        tx = tx if tx > tl else tl
        # tx必须大于tl，即不能左出界
        tx = tx if tx < in_size-tl else in_size-tl
        # 裁剪不能右超界，只能裁到头
        ty = ty if ty > tl else tl
        #类似tx，修正ty，不能上出界
        ty = ty if ty < in_size-tl else in_size-tl
        # 不能下出界

        # 下面定义矩形四个角
        w_off = int(tx-tl) if (tx-tl) > 0 else 0
        h_off = int(ty-tl) if (ty-tl) > 0 else 0
        w_end = int(tx+tl) if (tx+tl) < in_size else in_size
        h_end = int(ty+tl) if (ty+tl) < in_size else in_size
        # 得到掩模矩阵mk
        mk = (h(x-w_off) - h(x-w_end)) * (h(y-h_off) - h(y-h_end))
        # 点乘，得到裁剪后的图片
        xatt = images[i] * mk

        # 只留下裁剪后的图像
        xatt_cropped = xatt[:, w_off:w_end, h_off:h_end]
        # 采样之前先扩维
        before_upsample = Variable(xatt_cropped.unsqueeze(0))
        # 采样，规范裁剪后图像的大小
        xamp = F.interpolate(before_upsample, size=(224,224), mode='bilinear', align_corners = True)
        # 再压缩，便于后期的拼接
        ret.append(xamp.data.squeeze())
    # 将batch_size所有的图像拼接起来，便于后续返回
    ret_tensor = torch.stack(ret)
    self.save_for_backward(images, ret_tensor)
    # 返回裁剪后的图像
    return ret_tensor

    损失函数¶

    尺度内分类损失，三个尺度求和

    $$ L_1=\sum^3_{s=1}L_{cls}(Y^{(s)},Y^*) $$
    交叉熵损失函数可以直接通过torch.nn.functional调用

    def multitask_loss(preds, labels):
    loss = []
    # 遍历批次(batch_size)
    for i in range(len(preds)):
        # 会得到三个损失，每个损失对应该尺度上的分类损失
        loss.append(F.cross_entropy(preds[i], labels))
    return loss

    尺度间成对排序损失

    $$ L_2=\sum^2_{s=1}{L_{rank}(p^{s}_t,p^{(s+1)}_t)} $$
    其中
    $$ L_{rank}(p^{s}_t,p^{(s+1)}_t)=max\{0,p^{s}_t-p^{(s+1)}_t+margin\} $$ 

    def pairwise_ranking_loss(preds, size_average = True):
    """
    输入preds：每个尺度对应labels的预测值
    如果有三个尺度，第二维度就是3
    preds = [logits1[class], logits2[class], logits2[class]]
    """
    if len(preds[0]) <= 1:
        # preds第一维度代表批量，第二维度代表尺度
        # 如果只有一个尺度，则不存在尺度间成对排序损失，返回零
        return torch.zeros(1).cuda()
    else:
        losses = []
        for pred in preds:
            # 按批量进行遍历
            loss = []
            for i in range(len(pred)-1):
                rank_loss = (pred[i]-pred[i+1] + 0.05).clamp(min = 0)
                # margin设为0.05
                # 前后两个尺度的预测概率做差，再加上阈值
                # clamp函数相当于将数组规定在某范围内
                # 这里代表rank_loss最小值如果小于0，则被从新赋值为0
                loss.append(rank_loss)
                # 两个尺度对损失求和，再并入总损失
            loss = torch.sum(torch.stack(loss))
            losses.append(loss)
            # 拼接起来，便于求和或求平均值
        losses = torch.stack(losses)
        if size_average:
            # 是否对损失求平均值
            losses = torch.mean(losses)
        else:
            losses = torch.sum(losses)
        # 最后返回损失
        return losses

    训练流程¶

    名词解释：¶

    • cls：特征提取网络和全连接层参数的优化过程
    • apn：ANP注意力区域定位参数的优化过程
    • cls_loss：cls训练过程的损失，对应尺度内分类损失
    • anp_loss：anp训练过程的损失，对应尺度间排序损失
    • cls_iter：cls上训练次数，即参数更新次数(每个batch代表训练一次)
    • cls_epoch：cls上数据集遍历次数
    • cls_steps：cls优化器的学习率更新次数
    • old_cls_loss：上一次训练，cls的损失(旧损失)
    • new_cls_loss：本次训练，cls的损失(新损失)(用于计算损失变化)
    • cls_tol：cls损失无变化的次数(变化小于某阈值)

    将cls换成anp，形成的名词与原来的名词代表含义类似，只是换了个优化分支，换到了APN的优化这一分支

    • iteration：总的训练次数(参数更新次数)
    • switch_step：两个过程的交替次数

    APN的预训练¶

    预训练过程

    def pretrainAPN(trainset, trainloader):
    epoch_size = len(trainset) // 4
    # 这里的4代表batch_size=4，可以替换
    apn_steps, apn_epoch = 1, -1
    # 将trainloader生成迭代器
    batch_iterator = iter(trainloader)
    for _iter in range(0, 20000):
        iteration = _iter
        # 当前的更新次数
        if (not batch_iterator) or (iteration % epoch_size == 0):
            # 代表遍历了一个batch
            batch_iterator = iter(trainloader)

        if iteration % epoch_size == 0:
            # 遍历了一个batch时，记录次数，通过该次数更新学习率
            apn_epoch += 1
            if apn_epoch in decay_steps:
                # 如果到了设定好的次数，就更新一次
                apn_steps += 1
                adjust_learning_rate(opt2, 0.1, apn_steps, args.lr)
        # 读取图片和标签
        images, labels = next(batch_iterator)
        images, labels = Variable(images, requires_grad = True), Variable(labels)
        if args.cuda:
            images, labels = images.cuda(), labels.cuda()

        t0 = time.time()
        # 预测，得到特征图和注意力区域定位点
        _, conv5s, attens, _ = net(images)

        opt2.zero_grad()
        # 在特征图中寻找响应值最高的区域
        weak_loc = []
        for i in range(len(conv5s)):
            # 两个尺度的特征图
            loc_label = torch.ones([images.size(0),3]) * 0.33 
            # (批量，定位点)
            # 0.33代表归一化后的边长，loc_label具体意义见下面代码
            # tl = 0.25, fixed
            resize = 448
            if i >= 1:
                resize = 224
            if args.cuda:
                loc_label = loc_label.cuda()
            for j in range(images.size(0)):
                # batch遍历
                # 第i尺度的j批量
                response_map = conv5s[i][j]
                # 上采样放大
                response_map = F.interpolate(response_map.unsqueeze(0), size = [resize, resize]).squeeze(0)
                # 对特征图按第一维度求均值(每张图都求一次均值)
                # 得到448*448的矩阵
                response_map = response_map.mean(0)
                # 特征图先转化成向量，然后得到最大值(响应值最大)的位置(即索引)
                rawmaxidx = response_map.view(-1).max(0)[1]

                idx = []
                # 前面得到了响应值处最大的索引，但只是一个序号值
                # 下面通过运算，将序号转化成两点坐标
                for d in list(response_map.size())[::-1]:
                    # d:448->448，分别代表列宽和行高
                    idx.append(rawmaxidx % d)
                    rawmaxidx = rawmaxidx / d
                # 由于ANP得到的坐标就是归一化之后的坐标(最后经过了sigmoid)
                # 因此，这里也需要将得到的坐标进行归一化
                loc_label[j, 0] = (idx[1].float() + 0.5) / response_map.size(0)
                loc_label[j, 1] = (idx[0].float() + 0.5) / response_map.size(1)
                # loc_label里的第一二个值分别代表响应值最高的横坐标和纵坐标
                # 第三个值前面已经定好，为0.33，代表正方形边长的一半
            # loc_label里面的三个值代表了响应值高的区域
            weak_loc.append(loc_label)
        # smooth_l1_loss为平滑后的绝对值损失函数
        # loss1和loss2分别代表两个尺度的损失(坐标偏差)
        weak_loss1 = F.smooth_l1_loss(attens[0], weak_loc[0])
        weak_loss2 = F.smooth_l1_loss(attens[1], weak_loc[1])
        # 让预测的attens所在的区域，和经过VGG特征提取后响应最大的区域(长宽均为原来的一半)做损失对比
        # 即，让attens趋于VGG特征提取后网络关注的部分
        apn_loss = weak_loss1 + weak_loss2
        apn_loss.backward()
        opt2.step()
        # 参数更新
        t1 = time.time()

        if (iteration % 20) == 0:
            print(" [*] pre_apn_epoch[%d], || pre_apn_iter %d || pre_apn_loss: %.4f || Timer: %.4fsec"%(apn_epoch, iteration, apn_loss.item(), (t1 - t0)))

        logger.scalar_summary('pre_apn_loss', apn_loss.item(), iteration + 1)

    return 20000, apn_epoch, apn_steps
# 20000代表训练了20000次

    交替训练¶

    紧跟APN的预训练，首先初始化参数

    # cls_iter是训练次数，cls_epoch是数据集

    接下来进行交替，当分类损失和注意力定位损失不变(变化小于给定的阈值)时，停止训练。两组参数训练优化的过程类似，差别就在于损失的计算方法不同以及使用的优化器不同，因此下文只对cls的训练过程加以注释说明。

    尺度内分类损失：只需要得到各个尺度的预测概率，然后计算交叉熵损失 尺度间成对排序损失：先提取每个尺度的预测结果中，label的概率，然后再将不同尺度的概率结果加以计算，得到损失，最终用来优化

        opt1.zero_grad()
        # 分别计算三个尺度最后预测结果的尺度间分类损失
        new_cls_losses = multitask_loss(logits, labels)
        # 求和，得到总损失
        new_cls_loss = sum(new_cls_losses)
        # 反向传播，计算梯度
        new_cls_loss.backward()
        # 更新参数
        opt1.step()
        t1 = time.time()
        # 比较损失是否有变化
        # cls_tol代表cls损失连续未变化的次数

    # 读数据集+预测

    def test(testloader, iteration):
    # 模型调成测试过程
    net.eval()
    with torch.no_grad():
        # corrects分别为3个尺度正确预测的个数
        corrects1 = 0
        corrects2 = 0
        corrects3 = 0
        # cnt为测试集图片总数
        cnt = 0
        # 记录损失
        test_cls_losses = []
        test_apn_losses = []
        # 读取数据
        for test_images, test_labels in testloader:
            if args.cuda:
                test_images = test_images.cuda()
                test_labels = test_labels.cuda()
            cnt += test_labels.size(0)
            # 预测，得到概率
            logits, _, _, _ = net(test_images)

            preds = []
            # 计算尺度内分类损失以及尺度间排序损失
            for i in range(len(test_labels)):
                pred = [logit[i][test_labels[i]] for logit in logits]
                preds.append(pred)
            test_cls_losses = multitask_loss(logits, test_labels)
            test_apn_loss = pairwise_ranking_loss(preds)
            test_cls_losses.append(sum(test_cls_losses))
            test_apn_losses.append(test_apn_loss)
            # 分别得到预测值，再和label做比较，得到正确预测的个数
            _, predicted1 = torch.max(logits[0], 1)
            correct1 = (predicted1 == test_labels).sum()
            corrects1 += correct1
            _, predicted2 = torch.max(logits[1], 1)
            correct2 = (predicted2 == test_labels).sum()
            corrects2 += correct2
            _, predicted3 = torch.max(logits[2], 1)
            correct3 = (predicted3 == test_labels).sum()
            corrects3 += correct3
        # 最后损失做平均
        test_cls_losses = torch.stack(test_cls_losses).mean()
        test_apn_losses = torch.stack(test_apn_losses).mean()
        # 计算准确率
        accuracy1 = corrects1.float() / cnt
        accuracy2 = corrects2.float() / cnt
        accuracy3 = corrects3.float() / cnt
        # 写入日志
        logger.scalar_summary('test_cls_loss', test_cls_losses.item(), iteration + 1)
        logger.scalar_summary('test_rank_loss', test_apn_losses.item(), iteration + 1)
        logger.scalar_summary('test_acc1', accuracy1.item(), iteration + 1)
        logger.scalar_summary('test_acc2', accuracy2.item(), iteration + 1)
        logger.scalar_summary('test_acc3', accuracy3.item(), iteration + 1)
        print(" [*] Iter %d || Test accuracy1: %.4f, Test accuracy2: %.4f, Test accuracy3: %.4f"%(iteration, accuracy1.item(), accuracy2.item(), accuracy3.item()))
    # 结束测试，模型调成训练过程
    net.train()

def tensor_to_img(x, imtype=np.uint8):
   # 图像的均值和方差，用于反标准化，可以修改
   mean = [109.97 / 255., 127.34 / 255., 123.88 / 255.]
   std = [1. / 255., 1. / 255., 1. / 255.]

   if not isinstance(x, np.ndarray):
      if isinstance(x, torch.Tensor):  # get the data from a variable
         image_tensor = x.data
      else:
         return x
      image_numpy = image_tensor.cpu().float().numpy()  # convert it into a numpy array
      if image_numpy.shape[0] == 1:  # grayscale to RGB
         image_numpy = np.tile(image_numpy, (3, 1, 1))
      for i in range(len(mean)):
         # 反标准化(标准化的反运算)
         image_numpy[i] = image_numpy[i] * std[i] + mean[i]
      # 再乘回到255的标准量纲，返回到正常图像
      image_numpy = image_numpy * 255
      # 交换数组的维度，0,1,2->1,2,0，即3,448,448->448,448,3(转化成cv2的格式)
      image_numpy = np.transpose(image_numpy, (1, 2, 0))  # post-processing: tranpose and scaling
   else:  # if it is a numpy array, do nothing
      image_numpy = x
   return image_numpy.astype(imtype)
