    网络初始化阶段¶

    def __init__(self, num_classes, M=32, net='inception_mixed_6e', pretrained=False):
    # 分类数目，M：attention mps的数量，net：特征提取网络，pretrained是否预训练
    super(WSDAN, self).__init__()
    # 特征提取网络的定义
    self.num_classes = num_classes
    self.M = M
    self.net = net
    # net为预先传入的特征提取层字符串名称
    # Network Initialization
    # 网络模型的选择
    if 'inception' in net:
        if net == 'inception_mixed_6e':
            # 读取并调用inception_v3的预训练模型。使用与训练的mixed_6e层获取特征。
            self.features = inception_v3(pretrained=pretrained).get_features_mixed_6e()
            self.num_features = 768
        elif net == 'inception_mixed_7c':
            self.features = inception_v3(pretrained=pretrained).get_features_mixed_7c()
            self.num_features = 2048
        else:
            raise ValueError('Unsupported net: %s' % net)
    elif 'vgg' in net:
        self.features = getattr(vgg, net)(pretrained=pretrained).get_features()
        self.num_features = 512
    elif 'resnet' in net:
        self.features = getattr(resnet, net)(pretrained=pretrained).get_features()
        self.num_features = 512 * self.features[-1][-1].expansion
        # self.num_features为输出的特征图通道数量，此时位2048
    else:
        raise ValueError('Unsupported net: %s' % net)

    # Attention Maps
    # 卷积核大小为1*1的卷积层意义就是改变通道数，在这里，经过卷积、标准化、ReLU改变后的特征图我们命名为注意力图
    # BasicConv2d的定义见注意力图的生成
    self.attentions = BasicConv2d(self.num_features, self.M, kernel_size=1)

    # 双线性池化，这里以全局平均为例
    # BAP的定义见双线性池化
    self.bap = BAP(pool='GAP')

    # 全连接层的定义，输入大小为注意力图数量M乘以特征提取后的通道数，输出大小为种类数量
    self.fc = nn.Linear(self.M * self.num_features, self.num_classes, bias=False)
    # 输出日志
    logging.info('WSDAN: using {} as feature extractor, num_classes: {}, num_attentions: {}'.format(net, self.num_classes, self.M))

    前向传播阶段¶

    def forward(self, x):
    # 前向传播
    batch_size = x.size(0)
    # 先得到特征图，768*26*26
    feature_maps = self.features(x)
    # 再得到注意力图
    if self.net != 'inception_mixed_7c':
        attention_maps = self.attentions(feature_maps)# 32*26*26
    else:
        attention_maps = feature_maps[:, :self.M, ...]
    # 通过特征图与注意力图生成feature_matrix
    feature_matrix = self.bap(feature_maps, attention_maps)

    # 将feature_matrix传入全连接进行预测
    p = self.fc(feature_matrix * 100.)

    # Generate Attention Map
    if self.training:
        # 随机选择注意力图，进行下一步操作
        # Randomly choose one of attention maps Ak
        attention_map = []
        for i in range(batch_size):
            # 一组batch中，每张图按顺序生成
            attention_weights = torch.sqrt(attention_maps[i].sum(dim=(1, 2)).detach() + EPSILON)
            # 先将注意力图全局求和(分别沿1,2维度求和)，逐元素对张量进行开根运算
            attention_weights = F.normalize(attention_weights, p=1, dim=0)
            # 进一步进行，归一化得到注意力图的权重
            k_index = np.random.choice(self.M, 2, p=attention_weights.cpu().numpy())
            # 这里的参数2代表回传两张注意力图
            # 按概率p，随机在32内选俩值(权重越大，被选的概率越高)，得到索引
            # p是概率，通过对图片像素加权的平方得到。
            attention_map.append(attention_maps[i, k_index, ...])
        # 最后将注意力图堆叠
        attention_map = torch.stack(attention_map)

    else:
        #val或test时，对所有注意力图求均值
        attention_map = torch.mean(attention_maps, dim=1, keepdim=True)  # (B, 1, H, W)

    return p, feature_matrix, attention_map

    注意力图的生成¶

    class BasicConv2d(nn.Module):
    # 在网络中输出的feature map 的尺寸为 B*C*H*W，经过处理后，尺寸变为B*M*H*W，其中M为预先定义好的注意力图的数量。
    def __init__(self, in_channels, out_channels, **kwargs):
        # 这里kernel_size为1(存在了kwargs,前面已经传入了)
        super(BasicConv2d, self).__init__()
        # 代表卷积核大小的参数存在于**kwargs中
        self.conv = nn.Conv2d(in_channels, out_channels, bias=False, **kwargs)
        # 归一化操作
        self.bn = nn.BatchNorm2d(out_channels, eps=0.001)

    def forward(self, x):
        x = self.conv(x)
        x = self.bn(x)
        return F.relu(x, inplace=True)
        # 归一化之后再经过relu，正好将显著性低的区域变为零，只保留显著性高的区域

    双线性注意力池化¶

    class BAP(nn.Module):
    def __init__(self, pool='GAP'):
        super(BAP, self).__init__()
        assert pool in ['GAP', 'GMP']#采样策略，GAP是全局平均，GMP是全局最大
        if pool == 'GAP':
            self.pool = None
        else:
            self.pool = nn.AdaptiveMaxPool2d(1)

    def forward(self, features, attentions):
        B, C, H, W = features.size()
        _, M, AH, AW = attentions.size()

        # match size
        if AH != H or AW != W:
            attentions = F.upsample_bilinear(attentions, size=(H, W))# 将注意力图调整成和特征图一样的大小尺寸
            # 如果attention maps和feature高和宽不一致，采用双线性插值调整。
        # feature_matrix: (B, M, C) -> (B, M * C)
        if self.pool is None:# 如果是全局平均
            feature_matrix = (torch.einsum('imjk,injk->imn', (attentions, features)) / float(H * W)).view(B, -1)
            # 把尺寸为i,m,j,k 的attention和尺寸为i,n,j,k的feature，做双线性池化
            # injk中i不变，n和jk交换位置（转置），之后做矩阵乘法（矩阵外积，非点乘），得到imn的矩阵
            # 得到B*M*C的三维矩阵。除以H*W后，用函数view变为B行，M*C列的二维矩阵
            # 做完这些变换后得到的feature_matrix，相当于把每个通道的图片展成一维矩阵，然后M个一维矩阵拼接组成一行。
            # feature_matrix的每一行是一张图片的所有特征。
        else:
            feature_matrix = []
            for i in range(M):
                AiF = self.pool(features * attentions[:, i:i + 1, ...]).view(B, -1)
                #首先，把M张尺寸为B*1*H*W的attention map, 依次和B*C*H*W的feature maps相乘。
                # 得到M份B*C*H*W 的part Feature maps。
                #接着，做自适应池化（AdaptiveAvgPool2d(1)），尺寸变为B*C*1*1。然后用函数view(B, -1) 变为大小为B*C的矩阵。
                # （x.view(batchsize, -1) 中batchsize指转换后有几行，而-1指在不告诉函数有多少列的情况下，
                # 根据原tensor数据和batchsize自动分配列数。）
                feature_matrix.append(AiF)
                #循环结束后feature_matrix的大小为M个B*C的矩阵。
            feature_matrix = torch.cat(feature_matrix, dim=1)
            #把feature_mareix 重新排列为B行，C*M列的矩阵。

        # sign-sqrt
        feature_matrix = torch.sign(feature_matrix) * torch.sqrt(torch.abs(feature_matrix) + EPSILON)
        #对上面的结果做处理。EPSILON = 1e-12 使根号下不为零。
        #这个处理是为了得到最显著的特征。
        # l2 normalization along dimension M and C
        feature_matrix = F.normalize(feature_matrix, dim=-1)#归一化。
        return feature_matrix

    损失函数¶

    注意力正则化损失，即：

    $$ L_A=\sum^M_{k=1}{||f_k-c_k||^2_2} $$ 

    # 具体实现过程
class CenterLoss(nn.Module):
    def __init__(self):
        super(CenterLoss, self).__init__()
        self.l2_loss = nn.MSELoss(reduction='sum')
    def forward(self, outputs, targets):
        # 传入输出的特征矩阵与特征中心
        return self.l2_loss(outputs, targets) / outputs.size(0)
        # 除以batch_size，得到每张图的平均损失

    训练过程¶

    def train(**kwargs):
    # **kwargs: **会以键/值对的形式解包一个字典，使其成为独立的关键字参数。
    # 读取各种参数
    logs = kwargs['logs']
    data_loader = kwargs['data_loader']
    net = kwargs['net']
    feature_center = kwargs['feature_center']
    optimizer = kwargs['optimizer']
    pbar = kwargs['pbar']

    #参数调用reset（）函数，回到最初设置（全零矩阵）。
    loss_container.reset()
    raw_metric.reset()
    crop_metric.reset()
    drop_metric.reset()

    # 开始训练
    start_time = time.time()
    net.train()
    for i, (X, y) in enumerate(data_loader):
        # enumerate() 函数用于将一个可遍历的数据对象(如列表、元组或字符串)
        # 组合为一个索引序列，同时列出数据和数据下标.
        optimizer.zero_grad()
        # optimizer.zero_grad()将梯度初始化为零。
        # X是训练图片，y是对应标签
        X = X.to(device)
        y = y.to(device)

        ##################################
        # Raw Image
        ##################################
        # 首先传入X，得到预测概率、特征矩阵以及注意力图
        y_pred_raw, feature_matrix, attention_map = net(X)

        # Update Feature Center
        feature_center_batch = F.normalize(feature_center[y], dim=-1)
        # 特征中心归一化，更新特征中心(feature_center)
        feature_center[y] += config.beta * (feature_matrix.detach() - feature_center_batch)
        # detach()返回一个新的Variable，从当前计算图中分离下来的，但是仍指向原变量的存放位置。
        # 作差之后，不断叠加变化量到feature_matrix

        ##################################
        # Attention Cropping
        ##################################
        with torch.no_grad():
            # torch.no_grad()不需要梯度
            # batch_augment的解读见基于注意力的数据增广
            crop_images = batch_augment(X, attention_map[:, :1, :, :], mode='crop', theta=(0.4, 0.6), padding_ratio=0.1)
            # crop中输入的attention map是attention_map[:, :1, :, :] attention maps的第一张图片。
        # crop只要得到一个特征的放大图，只需要找到min和max像素的位置，所以一张特征图足够。

        # 裁剪图像传入网络进行预测
        y_pred_crop, _, _ = net(crop_images)

        ##################################
        # Attention Dropping
        ##################################
        with torch.no_grad():
            drop_images = batch_augment(X, attention_map[:, 1:, :, :], mode='drop', theta=(0.2, 0.5))
            # drop中输入的attention map是attention_map[:, 1：, :, :] attention maps:除去第一张图片外的所有图片。(一般是1个，跟网络定的参数有关系)
        # drop是把所有小于阈值的像素置0，操作是像素级，用数量多的attention maps可以涵盖尽量大且准确的特征区,因此可以传入多张注意力图进行删除操作。

        # 注意力删除图像传入网络进行预测
        y_pred_drop, _, _ = net(drop_images)
        # batch_loss的计算采用交叉熵损失。考虑四个预测部分，结果相加。
        # loss
        batch_loss = cross_entropy_loss(y_pred_raw, y) / 3. + \
                     cross_entropy_loss(y_pred_crop, y) / 3. + \
                     cross_entropy_loss(y_pred_drop, y) / 3. + \
                     center_loss(feature_matrix, feature_center_batch)

        # backward
        #backward反向传播，optimizer.step() 根据网络反向传播的梯度来更新网络参数。
        batch_loss.backward()
        optimizer.step()

        # 计算损失和精度
        with torch.no_grad():
            epoch_loss = loss_container(batch_loss.item())
            epoch_raw_acc = raw_metric(y_pred_raw, y)
            epoch_crop_acc = crop_metric(y_pred_crop, y)
            epoch_drop_acc = drop_metric(y_pred_drop, y)
            # metrics ：epoch_loss采用AverageMeter，计算多个类的平均值。
            # epoch_raw，crop，drop采用TopKAccuracyMetric（选取概率最大的 k 个标签）计算平均值。

        # end of this batch 更新进度条
        batch_info = 'Loss {:.4f}, Raw Acc ({:.2f}, {:.2f}), Crop Acc ({:.2f}, {:.2f}), Drop Acc ({:.2f}, {:.2f})'.format(
            epoch_loss, epoch_raw_acc[0], epoch_raw_acc[1],
            epoch_crop_acc[0], epoch_crop_acc[1], epoch_drop_acc[0], epoch_drop_acc[1])
        pbar.update()
        pbar.set_postfix_str(batch_info)

    # 将训练精度与损失保存到日志里面
    logs['train_{}'.format(loss_container.name)] = epoch_loss
    logs['train_raw_{}'.format(raw_metric.name)] = epoch_raw_acc
    logs['train_crop_{}'.format(crop_metric.name)] = epoch_crop_acc
    logs['train_drop_{}'.format(drop_metric.name)] = epoch_drop_acc
    logs['train_info'] = batch_info
    end_time = time.time()

    # 输出训练信息
    logging.info('Train: {}, Time {:3.2f}'.format(batch_info, end_time - start_time))

            X = X.to(device)

            ##################################
            y_pred_raw, _, attention_map = net(X)

            ##################################
            # Object Localization and Refinement
            ##################################

            crop_images = batch_augment(X, attention_map, mode='crop', theta=0.1, padding_ratio=0.05)

            ##################################
            # Final prediction
            ##################################
             最后的预测值等于raw image的预测值和crop image的预测值。
            y_pred = (y_pred_raw + y_pred_crop) / 2.

            batch_loss = cross_entropy_loss(y_pred, y)

    def batch_augment(images, attention_map, mode='crop', theta=0.5, padding_ratio=0.1):
    batches, _, imgH, imgW = images.size()
    if mode == 'crop':
        # copy模式
        crop_images = []
        for batch_index in range(batches):
            # 每张图片分别进行裁剪
            atten_map = attention_map[batch_index:batch_index + 1]
            if isinstance(theta, tuple):
                theta_c = random.uniform(*theta) * atten_map.max()
                # .max()为最大值操作
                # theta为数组时，每次从theta给出的范围中随机生成一个实数。
                # uniform()方法为随机生成下一个实数。
            else:
                theta_c = theta * atten_map.max()
                # 阈值为theta * atten_map.max()
            # theta_c相当于得到一个阈值，大于该阈值的区域当做被裁剪区域
            crop_mask = F.upsample_bilinear(atten_map, size=(imgH, imgW)) >= theta_c
            # 对atten_map（每次循环，从attention maps 取1张图片作为atten_map）首先做双线性上采样，使其大小变为标准大小
            # 然后做一个阈值判断，大于阈值的部分置为true（1），小于阈值的部分置为false（0）。
            nonzero_indices = torch.nonzero(crop_mask[0, 0, ...])
            # 得到非零元素的索引，即需要裁剪的部分
            # crop_mask的第一个通道，非零值的位置。结果的第一列是图片的行号，第二列是列号。

            height_min = max(int(nonzero_indices[:, 0].min().item() - padding_ratio * imgH), 0)
            height_max = min(int(nonzero_indices[:, 0].max().item() + padding_ratio * imgH), imgH)
            width_min = max(int(nonzero_indices[:, 1].min().item() - padding_ratio * imgW), 0)
            width_max = min(int(nonzero_indices[:, 1].max().item() + padding_ratio * imgW), imgW)
            # 然后找到第一列的最大值和最小值的差值作为高，第二列的最大值与最小值的差值作为宽。
            # 两部分均向外扩张padding_ratio * imgH作为预留部分

            crop_images.append(
                F.upsample_bilinear(images[batch_index:batch_index + 1, :, height_min:height_max,                                               width_min:width_max], size=(imgH, imgW)))
            # 首先按得到的指标进行截取图像(截取图像数据大于阈值的部分)，然后再进行上采样放大，放缩成原图像大小
        crop_images = torch.cat(crop_images, dim=0)# 将裁剪好的图片拼接
        return crop_images
    elif mode == 'drop':
        # drop模式
        drop_masks = []
        for batch_index in range(batches):
            atten_map = attention_map[batch_index:batch_index + 1]
            if isinstance(theta, tuple):
                theta_d = random.uniform(*theta) * atten_map.max()
            else:
                theta_d = theta * atten_map.max()
           # 和裁剪类似，得到用于判断的阈值

            drop_masks.append(F.upsample_bilinear(atten_map, size=(imgH, imgW)) < theta_d)
            # 做双线性上采样，小于阈值置为0，大于置为1。
        drop_masks = torch.cat(drop_masks, dim=0)
        # 把drop_mask按列合并，变成B*（M-1）*H*W。
        drop_images = images * drop_masks.float()
        # drop_mask与image相乘，得到dropping后的图片。
        # 直接做点乘就可以(torch中的*就是点乘)
        return drop_images
    else:
        # 否则报错，数据增广策略必须是裁剪或者删除
        raise ValueError('Expected mode in [\'crop\', \'drop\'], but received unsupported augmentation method %s' % mode)

    # 保存策略定义，每次保存最好的模型
class ModelCheckpoint(Callback):
    def __init__(self, savepath, monitor='val_topk_accuracy', mode='max'):
        self.savepath = savepath
        # 模型保存地址
        self.monitor = monitor
        # 监控方式，监控测试精度
        self.mode = mode
        # 保存策略，默认保存精度最高的模型
        # 还有一种判断措施就是监控测试的损失，保存损失最小的模型
        self.reset()
        super(ModelCheckpoint, self).__init__()

    def reset(self):
        if self.mode == 'max':
            self.best_score = float('-inf')
            # 初始化最高精度为负无穷
        else:
            self.best_score = float('inf')

    def set_best_score(self, score):
        if isinstance(score, np.ndarray):
            self.best_score = score[0]
        else:
            self.best_score = score

    def on_epoch_begin(self):
        pass

    def on_epoch_end(self, logs, net, **kwargs):
        current_score = logs[self.monitor]
        # 得到当前的测试精度
        if isinstance(current_score, np.ndarray):
            current_score = current_score[0]

        if (self.mode == 'max' and current_score > self.best_score) or \
            (self.mode == 'min' and current_score < self.best_score):
            # 如果在测试过程中，测试精度比现有的最高精度都高，就需要更新最高的精度，并保存模型
            self.best_score = current_score

            if isinstance(net, torch.nn.DataParallel):
            # 如果是多GPU并行训练，则需要多加一层module
                state_dict = net.module.state_dict()
            else:
                state_dict = net.state_dict()

            for key in state_dict.keys():

                torch.save({
                    'logs': logs,
                    'state_dict': state_dict,
